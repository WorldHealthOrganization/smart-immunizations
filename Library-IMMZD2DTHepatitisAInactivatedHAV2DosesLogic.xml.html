 * Library: IMMZD2DTHepatitisAInactivatedHAV2DosesLogic (IMMZ.D2.DT.Hepatitis A.Inactivated HAV 2 doses)
 * Rule: Determine if the client is due for a hepatitis A vaccination according to the national immunization schedule
 * Decision Table: Inactivated hepatitis A virus (HAV), 2-dose schedule
 * Trigger: IMMZ.D2 Determine required vaccination(s) if any
 */
library IMMZD2DTHepatitisAInactivatedHAV2DosesLogic

using FHIR version '4.0.1'
include FHIRHelpers version '4.0.1'

include IMMZD2DTHepatitisAEncounterElements called Encounter

parameter Today Date default Today()

context Patient


/*
@dynamicValue: Guidance
*/
define &quot;Guidance&quot;:
  case 
    when &quot;Client is not due for Hepatitis A vaccination&quot; then &quot;Client is not due for Hepatitis A vaccination Guidance&quot;
    when &quot;Client is due for Hepatitis A vaccination&quot; then &quot;Client is due for Hepatitis A vaccination Guidance&quot;
    when &quot;Hepatitis A immunization schedule is complete&quot; then &quot;Hepatitis A immunization schedule is complete Guidance&quot; 
    else ''
  end
  
define &quot;Has Guidance&quot;:
  &quot;Guidance&quot; is not null and &quot;Guidance&quot; != ''

/*
@output: Client is not due for Hepatitis A vaccination Case 1
@pseudocode: &quot;Immunization recommendation status&quot; = &quot;Not due&quot;
*/
define &quot;Client is not due for Hepatitis A vaccination Case 1&quot;:
  Encounter.&quot;Client's age is less than 12 months&quot;

/*
@output: Client is not due for Hepatitis A vaccination Case 2
@pseudocode: &quot;Immunization recommendation status&quot; = &quot;Not due&quot;
*/
define &quot;Client is not due for Hepatitis A vaccination Case 2&quot;:
  Encounter.&quot;One hepatitis A primary series dose was administered&quot;
    and Encounter.&quot;The latest Hepatitis A dose was administered less than 6 months ago&quot;

/*
@output: Client is not due for Hepatitis A vaccination
@pseudocode: &quot;Immunization recommendation status&quot; = &quot;Not due&quot;
*/
define &quot;Client is not due for Hepatitis A vaccination&quot;:
  &quot;Client is not due for Hepatitis A vaccination Case 1&quot;
    or &quot;Client is not due for Hepatitis A vaccination Case 2&quot;

/*
@output: Client is not due for Hepatitis A vaccination Guidance
@guidance = Should not vaccinate client with first hepatitis A dose as client's age is less than 12 months. 
Check for any other vaccines due and inform the caregiver of when to come back for the first dose.
@guidance = Should not vaccinate client with second hepatitis A dose as the latest hepatitis A dose was administered less than 6 months ago. 
Check for any vaccines due and inform the caregiver of when to come back for the second dose.
*/
define &quot;Client is not due for Hepatitis A vaccination Guidance&quot;:
  case
    when &quot;Client is not due for Hepatitis A vaccination Case 1&quot; then 'Should not vaccinate client with first hepatitis A dose as client\'s age is less than 12 months. 
Check for any other vaccines due and inform the caregiver of when to come back for the first dose.'
    when &quot;Client is not due for Hepatitis A vaccination Case 2&quot; then 'Should not vaccinate client with second hepatitis A dose as the latest hepatitis A dose was administered less than 6 months ago. 
Check for any vaccines due and inform the caregiver of when to come back for the second dose.'
    else ''
  end

/*
@output: Client is due for Hepatitis A vaccination Case 1
@pseudocode: &quot;Immunization recommendation status&quot; = &quot;Due&quot;
*/
define &quot;Client is due for Hepatitis A vaccination Case 1&quot;:
  Encounter.&quot;No hepatitis A primary series doses were administered&quot;
    and Encounter.&quot;Client's age is more than or equal to 12 months&quot;

/*
@output: Client is due for Hepatitis A vaccination Case 2
@pseudocode: &quot;Immunization recommendation status&quot; = &quot;Due&quot;
*/
define &quot;Client is due for Hepatitis A vaccination Case 2&quot;:
  Encounter.&quot;One hepatitis A primary series dose was administered&quot;
    and Encounter.&quot;The latest Hepatitis A dose was administered more than 6 months ago&quot;

/*
@output: Client is due for Hepatitis A vaccination
@pseudocode: &quot;Immunization recommendation status&quot; = &quot;Due&quot;
*/
define &quot;Client is due for Hepatitis A vaccination&quot;:
  &quot;Client is due for Hepatitis A vaccination Case 1&quot;
    or &quot;Client is due for Hepatitis A vaccination Case 2&quot;

/*
@output: Client is due for Hepatitis A vaccination Guidance
@guidance = Should vaccinate client with first hepatitis A dose as no Hepatitis A doses were administered and client is within the appropriate age range. 
Check for contraindications.
@guidance = Should vaccinate client with second hepatitis A dose as the latest hepatitis A dose was administered more than 6 months ago. 
Check for contraindications.
*/
define &quot;Client is due for Hepatitis A vaccination Guidance&quot;:
  case
    when &quot;Client is due for Hepatitis A vaccination Case 1&quot; then 'Should vaccinate client with first hepatitis A dose as no Hepatitis A doses were administered and client is within the appropriate age range. 
Check for contraindications.'
    when &quot;Client is due for Hepatitis A vaccination Case 2&quot; then 'Should vaccinate client with second hepatitis A dose as the latest hepatitis A dose was administered more than 6 months ago. 
Check for contraindications.'
    else ''
  end

/*
@output: Hepatitis A immunization schedule is complete
@pseudocode: &quot;Immunization recommendation status&quot; = &quot;Complete&quot;
&quot;Completed the primary vaccination series&quot; = TRUE (where &quot;Vaccine type&quot; = &quot;Hepatitis A-containing vaccines&quot;)
*/
define &quot;Hepatitis A immunization schedule is complete&quot;:
  Encounter.&quot;Two hepatitis A primary series doses were administered&quot;

/*
@output: Hepatitis A immunization schedule is complete Guidance
@guidance: Hepatitis A immunization schedule is complete. Two hepatitis A primary series doses were administered. 
Check for any other vaccines due.
*/
define &quot;Hepatitis A immunization schedule is complete Guidance&quot;:
  'Hepatitis A immunization schedule is complete. Two hepatitis A primary series doses were administered. 
Check for any other vaccines due.'


/*
@test: Test expected results based on example patients
*/
define &quot;Test Validation&quot;:
  case

Check for any other vaccines due and inform the caregiver of when to come back for the first dose.'
    when Patient.id = 'HepatitisA09.2' then &quot;Client is due for Hepatitis A vaccination Case 1&quot; and &quot;Guidance&quot; = 'Should vaccinate client with first hepatitis A dose as no Hepatitis A doses were administered and client is within the appropriate age range. 
Check for contraindications.'
    when Patient.id = 'HepatitisA10.2' then &quot;Client is not due for Hepatitis A vaccination Case 2&quot; and &quot;Guidance&quot; = 'Should not vaccinate client with second hepatitis A dose as the latest hepatitis A dose was administered less than 6 months ago. 
Check for any vaccines due and inform the caregiver of when to come back for the second dose.'
    when Patient.id = 'HepatitisA11.2' then &quot;Client is due for Hepatitis A vaccination Case 2&quot; and &quot;Guidance&quot; = 'Should vaccinate client with second hepatitis A dose as the latest hepatitis A dose was administered more than 6 months ago. 
Check for contraindications.'
    when Patient.id = 'HepatitisA12.1' then &quot;Hepatitis A immunization schedule is complete&quot; and &quot;Guidance&quot; = 'Hepatitis A immunization schedule is complete. Two hepatitis A primary series doses were administered. 
Check for any other vaccines due.'
    else 'No test case set'
  end

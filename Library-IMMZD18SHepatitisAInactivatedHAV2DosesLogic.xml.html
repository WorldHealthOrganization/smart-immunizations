 * Library: IMMZD18SHepatitisAInactivatedHAV2DosesLogic (IMMZ.D18.S.Hepatitis A.Inactivated HAV 2-dose schedule)
 * Schedule Table: Inactivated hepatitis A virus (HAV), 2-dose schedule
 */
library IMMZD18SHepatitisAInactivatedHAV2DosesLogic

using FHIR version '4.0.1'
include FHIRHelpers version '4.0.1'

include WHOCommon called WC

include IMMZCommon called Common
include IMMZConcepts called Concepts

include IMMZEncounterElements called IE
include IMMZD2DTHepatitisAEncounterElements called Encounter

parameter Today Date default Today()


context Patient


/*
@output: Hepatitis A dose 1
@description: Provision of the hepatitis A dose 1
@trigger: Child's birth
@pseudo: 
@triggerDate: &quot;Date of birth&quot;
*/
define &quot;Hepatitis A dose 1&quot;:
  Encounter.&quot;No hepatitis A primary series doses were administered&quot;
  and not &quot;First hepatitis A dose from the primary series was administered&quot;


/*
@output: Hepatitis A dose 1 Create
@create: With a 2-dose schedule, the first dose should be given from ≥ 12 months of age.
*/
define &quot;Hepatitis A dose 1 Create&quot;:
  if &quot;Hepatitis A dose 1&quot;
  then 'With a 2-dose schedule, the first dose should be given from ≥ 12 months of age.' + '
Due Date: ' + ToString(&quot;Hepatitis A dose 1 Due Date&quot;)
  else ''


/*
@dynamicValue: Hepatitis A dose 1 Due Date
@pseudocode: &quot;Date of birth&quot; + 12 months
*/
define &quot;Hepatitis A dose 1 Due Date&quot;:
  if &quot;Hepatitis A dose 1&quot; then Patient.birthDate + 12 months
  else null


/*
@dynamicValue: Hepatitis A dose 1 Overdue
@pseudocode: To be determined by Member States; however, there is no recommended overdue date and individuals are always eligible to be vaccinated.
*/
define &quot;Hepatitis A dose 1 Overdue&quot;:
  null


/*
@dynamicValue: Hepatitis A dose 1 Expiration
@pseudocode: To be determined by Member States; however, there is no recommended expiration date and individuals are always eligible to be vaccinated.
*/
define &quot;Hepatitis A dose 1 Expiration&quot;:
  null

  
/*
@complete: First hepatitis A dose from the primary series was administered
@pseudocode: Count of vaccines administered (where &quot;Vaccine type&quot; = &quot;Hepatitis A-containing vaccines&quot; and &quot;Type of dose&quot; = &quot;Primary series&quot;) = 1
*/
define &quot;First hepatitis A dose from the primary series was administered&quot;:
  Encounter.&quot;One hepatitis A primary series dose was administered&quot;



/*
@output: Hepatitis A dose 2
@description: Provision of the hepatitis A dose 2
@trigger: First hepatitis A dose from the primary series was administered
@pseudo: Count of vaccines administered (where &quot;Vaccine type&quot; = &quot;Hepatitis A-containing vaccines&quot; and &quot;Type of dose&quot; = &quot;Primary series&quot;) = 1
@triggerDate: &quot;Date and time of vaccination&quot; (where &quot;Vaccine type&quot; = &quot;Hepatitis A-containing vaccines&quot;)
*/
define &quot;Hepatitis A dose 2&quot;:
  &quot;First hepatitis A dose from the primary series was administered&quot;
  and not &quot;Second hepatitis A dose from the primary series was administered. The primary series has been completed&quot;


/*
@output: Hepatitis A dose 2 Create
@create: The interval between doses is flexible, from 6 months to 4–5 years or more, but is usually 6–18 months.
*/
define &quot;Hepatitis A dose 2 Create&quot;:
  if &quot;Hepatitis A dose 2&quot;
  then 'The interval between doses is flexible, from 6 months to 4–5 years or more, but is usually 6–18 months.' + '
Due Date: ' + ToString(&quot;Hepatitis A dose 2 Due Date&quot;)
  else ''


/*
@dynamicValue: Hepatitis A dose 2 Due Date
@pseudocode: &quot;Date and time of vaccination&quot; (where &quot;Vaccine type&quot; = &quot;Hepatitis A-containing vaccines&quot;) + 6 months
*/
define &quot;Hepatitis A dose 2 Due Date&quot;:
  if &quot;Hepatitis A dose 2&quot; then Encounter.&quot;Date of Latest Hepatitis A Dose&quot; + 6 months
  else null


/*
@dynamicValue: Hepatitis A dose 2 Overdue
@pseudocode: To be determined by Member States; however, there is no recommended overdue date and individuals are always eligible to be vaccinated.
*/
define &quot;Hepatitis A dose 2 Overdue&quot;:
  null


/*
@dynamicValue: Hepatitis A dose 2 Expiration
@pseudocode: To be determined by Member States; however, there is no recommended expiration date and individuals are always eligible to be vaccinated.
*/
define &quot;Hepatitis A dose 2 Expiration&quot;:
  null

  
/*
@complete: Second hepatitis A dose from the primary series was administered. The primary series has been completed
@pseudocode: &quot;Completed the primary vaccination series&quot; = TRUE (where &quot;Vaccine type&quot; = &quot;Hepatitis A-containing vaccines&quot;)
*/
define &quot;Second hepatitis A dose from the primary series was administered. The primary series has been completed&quot;:
  Encounter.&quot;Two hepatitis A primary series doses were administered&quot;



/*
@test: Test expected results based on example patients
*/
define &quot;Test Validation&quot;:
  case
    when Patient.id = 'HepatitisA08.1' then &quot;Hepatitis A dose 1&quot;
    when Patient.id = 'HepatitisA09.2' then &quot;Hepatitis A dose 1&quot;
    when Patient.id = 'HepatitisA10.2' then &quot;Hepatitis A dose 2&quot;
    when Patient.id = 'HepatitisA11.2' then &quot;Hepatitis A dose 2&quot;
    when Patient.id = 'HepatitisA12.1' then &quot;Second hepatitis A dose from the primary series was administered. The primary series has been completed&quot;
    else 'No test case set'
  end

 * Library: IMMZD18SJEInactivatedVeroCellDerivedVaccineLogic (IMMZ.D18.S.JE.Inactivated Vero cell-derived vaccination schedule)
 * Schedule Table: Inactivated Vero cell-derived vaccination schedule (2-dose scheme)
 */
library IMMZD18SJEInactivatedVeroCellDerivedVaccineLogic

using FHIR version '4.0.1'
include FHIRHelpers version '4.0.1'

include WHOCommon called WC

include IMMZCommon called Common
include IMMZConcepts called Concepts

include IMMZEncounterElements called IE
include IMMZD2DTJEEncounterElements called Encounter

parameter Today Date default Today()


context Patient


/*
@output: Japanese encephalitis (JE) dose 1
@description: Provision of the JE dose 1
@trigger: Child's birth
@pseudo: 
@triggerDate: &quot;Date of birth&quot;
*/
define &quot;Japanese encephalitis (JE) dose 1&quot;:
  Encounter.&quot;No JE primary series doses were administered&quot;


/*
@output: Japanese encephalitis (JE) dose 1 Create
@create: JE dose 1 should be provided if the client's age is 6 months or older and the country is administering the Inactivated Vero cell-derived antigen.
*/
define &quot;Japanese encephalitis (JE) dose 1 Create&quot;:
  if &quot;Japanese encephalitis (JE) dose 1&quot;
  then 'JE dose 1 should be provided if the client\'s age is 6 months or older and the country is administering the Inactivated Vero cell-derived antigen.' + '
Due Date: ' + ToString(&quot;Japanese encephalitis (JE) dose 1 Due Date&quot;)
  else ''


/*
@dynamicValue: Japanese encephalitis (JE) dose 1 Due Date
@pseudocode: &quot;Date of birth&quot;+ 6 months
*/
define &quot;Japanese encephalitis (JE) dose 1 Due Date&quot;:
  if &quot;Japanese encephalitis (JE) dose 1&quot; then Patient.birthDate + 6 months
  else null


/*
@dynamicValue: Japanese encephalitis (JE) dose 1 Overdue
@pseudocode: To be determined by Member States; however, there is no recommended overdue date and individuals are always eligible to be vaccinated.
*/
define &quot;Japanese encephalitis (JE) dose 1 Overdue&quot;:
  null


/*
@dynamicValue: Japanese encephalitis (JE) dose 1 Expiration
@pseudocode: To be determined by Member States; however, there is no recommended expiration date and individuals are always eligible to be vaccinated.
*/
define &quot;Japanese encephalitis (JE) dose 1 Expiration&quot;:
  null

  
/*
@complete: One JE primary series dose was administered
@pseudocode: Count of vaccines administered (where &quot;Vaccine type&quot; = &quot;JE vaccines&quot; and &quot;Type of dose&quot; = &quot;Primary series&quot;) = 1
*/
define &quot;One JE primary series dose was administered&quot;:
  Encounter.&quot;One JE primary series dose was administered&quot;



/*
@output: JE dose 2
@description: Provision of the JE dose 2
@trigger: One JE primary series dose was administered
@pseudo: Count of vaccines administered (where &quot;Vaccine type&quot; = &quot;JE vaccines&quot; and &quot;Type of dose&quot; = &quot;Primary series&quot;) = 1
@triggerDate: &quot;Date and time of vaccination&quot; where &quot;Vaccine type&quot; = &quot;JE vaccines&quot;
*/
define &quot;JE dose 2&quot;:
  &quot;One JE primary series dose was administered&quot;
  and not &quot;Second JE dose from the primary series was administered. The primary series has been completed&quot;


/*
@output: JE dose 2 Create
@create: JE inactivated Vero cell-derived vaccine dose 2 should be provided if dose 1 was administered at least 4 weeks ago.
*/
define &quot;JE dose 2 Create&quot;:
  if &quot;JE dose 2&quot;
  then 'JE inactivated Vero cell-derived vaccine dose 2 should be provided if dose 1 was administered at least 4 weeks ago.' + '
Due Date: ' + ToString(&quot;JE dose 2 Due Date&quot;)
  else ''


/*
@dynamicValue: JE dose 2 Due Date
@pseudocode: &quot;Date and time of vaccination&quot; (where &quot;Vaccine type&quot; = &quot;JE vaccines&quot;) + 4 weeks
*/
define &quot;JE dose 2 Due Date&quot;:
  if &quot;JE dose 2&quot; then Encounter.&quot;Date of Latest JE Dose&quot; + 4 weeks
  else null


/*
@dynamicValue: JE dose 2 Overdue
@pseudocode: To be determined by Member States; however, there is no recommended overdue date and individuals are always eligible to be vaccinated.

  null

@dynamicValue: JE dose 2 Expiration

*/
define &quot;JE dose 2 Expiration&quot;:
  null

  
/*

*/
define &quot;Second JE dose from the primary series was administered. The primary series has been completed&quot;:
  Encounter.&quot;Two JE primary series doses were administered&quot;



/*

define &quot;Test Validation&quot;:
  case

    when Patient.id = 'JE10.2' then &quot;JE dose 2&quot;
    when Patient.id = 'JE11.2' then &quot;JE dose 2&quot;
    when Patient.id = 'JE12.1' then &quot;Second JE dose from the primary series was administered. The primary series has been completed&quot;
    else 'No test case set'
  end

 * Library: IMMZD2DTHib3DosesLogic (IMMZ.D2.DT.Hib.3 doses)
 * Rule: Determine if the client is due for a Haemophilus influenzae type b (Hib) vaccination according to the national immunization schedule
 * Decision Table: 3 primary doses without a booster dose (3p)
 * Trigger: IMMZ.D2 Determine required vaccination(s) if any
 */
library IMMZD2DTHib3DosesLogic

using FHIR version '4.0.1'
include FHIRHelpers version '4.0.1'

include IMMZD2DTHibEncounterElements called Encounter

parameter Today Date default Today()

context Patient


/*
@dynamicValue: Guidance
*/
define &quot;Guidance&quot;:
  case 
    when &quot;Client is not due for Hib vaccination&quot; then &quot;Client is not due for Hib vaccination Guidance&quot;
    when &quot;Client is due for Hib vaccination&quot; then &quot;Client is due for Hib vaccination Guidance&quot;
    when &quot;Hib immunization schedule is complete&quot; then &quot;Hib immunization schedule is complete Guidance&quot; 
    else ''
  end
  
define &quot;Has Guidance&quot;:
  &quot;Guidance&quot; is not null and &quot;Guidance&quot; != ''

/*
@output: Client is not due for Hib vaccination Case 1
@pseudocode: &quot;Immunization recommendation status&quot; = &quot;Not due&quot;
*/
define &quot;Client is not due for Hib vaccination Case 1&quot;:
  Encounter.&quot;Client's age is less than 6 weeks&quot;

/*
@output: Client is not due for Hib vaccination Case 2
@pseudocode: &quot;Immunization recommendation status&quot; = &quot;Not due&quot;
*/
define &quot;Client is not due for Hib vaccination Case 2&quot;:
  Encounter.&quot;One Hib primary series dose was administered&quot;
    and Encounter.&quot;Client's age is at least 6 weeks and not more than 5 years&quot;
    and Encounter.&quot;Client's age was less than 1 year when first Hib dose was administered&quot;
    and Encounter.&quot;The latest Hib dose was administered less than 4 weeks ago&quot;

/*
@output: Client is not due for Hib vaccination Case 3
@pseudocode: &quot;Immunization recommendation status&quot; = &quot;Not due&quot;
*/
define &quot;Client is not due for Hib vaccination Case 3&quot;:
  Encounter.&quot;Two Hib primary series doses were administered&quot;
    and Encounter.&quot;Client's age is between 6 weeks and not more than 5 years&quot;
    and Encounter.&quot;The latest Hib dose was administered less than 4 weeks ago&quot;

/*
@output: Client is not due for Hib vaccination Case 4
@pseudocode: &quot;Immunization recommendation status&quot; = &quot;Not due&quot;
*/
define &quot;Client is not due for Hib vaccination Case 4&quot;:
  Encounter.&quot;Client's age is more than 5 years&quot;

/*
@output: Client is not due for Hib vaccination
@pseudocode: &quot;Immunization recommendation status&quot; = &quot;Not due&quot;
*/
define &quot;Client is not due for Hib vaccination&quot;:
  &quot;Client is not due for Hib vaccination Case 1&quot;
    or &quot;Client is not due for Hib vaccination Case 2&quot;
    or &quot;Client is not due for Hib vaccination Case 3&quot;
    or &quot;Client is not due for Hib vaccination Case 4&quot;

/*
@output: Client is not due for Hib vaccination Guidance
@guidance = Should not vaccinate client with first Hib dose as client's age is less than 6 weeks. 
Check for any other vaccines due and inform the caregiver of when to come back for first dose.
@guidance = Should not vaccinate client with second Hib dose as latest Hib dose was administered less than 4 weeks ago. 
Check for any other vaccines due and inform the caregiver of when to come back for second dose.
@guidance = Should not vaccinate client with third Hib dose as latest Hib dose was administered less than 4 weeks ago. 
Check for any other vaccines due and inform the caregiver of when to come back for third dose.
@guidance = Should not vaccinate client with Hib dose as client is more than 5 years old and Hib vaccination is not required for healthy children aged over 5 years. 
Check for any other vaccines due.
*/
define &quot;Client is not due for Hib vaccination Guidance&quot;:
  case
    when &quot;Client is not due for Hib vaccination Case 1&quot; then 'Should not vaccinate client with first Hib dose as client\'s age is less than 6 weeks. 
Check for any other vaccines due and inform the caregiver of when to come back for first dose.'
    when &quot;Client is not due for Hib vaccination Case 2&quot; then 'Should not vaccinate client with second Hib dose as latest Hib dose was administered less than 4 weeks ago. 
Check for any other vaccines due and inform the caregiver of when to come back for second dose.'
    when &quot;Client is not due for Hib vaccination Case 3&quot; then 'Should not vaccinate client with third Hib dose as latest Hib dose was administered less than 4 weeks ago. 
Check for any other vaccines due and inform the caregiver of when to come back for third dose.'
    when &quot;Client is not due for Hib vaccination Case 4&quot; then 'Should not vaccinate client with Hib dose as client is more than 5 years old and Hib vaccination is not required for healthy children aged over 5 years. 
Check for any other vaccines due.'
    else ''
  end

/*
@output: Client is due for Hib vaccination Case 1
@pseudocode: &quot;Immunization recommendation status&quot; = &quot;Due&quot;
*/
define &quot;Client is due for Hib vaccination Case 1&quot;:
  Encounter.&quot;No Hib primary series doses were administered&quot;
    and Encounter.&quot;Client's age is between 6 weeks and 5 years&quot;

/*
@output: Client is due for Hib vaccination Case 2
@pseudocode: &quot;Immunization recommendation status&quot; = &quot;Due&quot;
*/
define &quot;Client is due for Hib vaccination Case 2&quot;:
  Encounter.&quot;One Hib primary series dose was administered&quot;
    and Encounter.&quot;Client's age is at least 6 weeks and not more than 5 years&quot;
    and Encounter.&quot;Client's age was less than 1 year when first Hib dose was administered&quot;
    and Encounter.&quot;The latest Hib dose was administered more than 4 weeks ago&quot;

/*
@output: Client is due for Hib vaccination Case 3
@pseudocode: &quot;Immunization recommendation status&quot; = &quot;Due&quot;
*/

  Encounter.&quot;Two Hib primary series doses were administered&quot;
    and Encounter.&quot;Client's age is between 6 weeks and not more than 5 years&quot;
    and Encounter.&quot;The latest Hib dose was administered more than 4 weeks ago&quot;

/*
@output: Client is due for Hib vaccination
@pseudocode: &quot;Immunization recommendation status&quot; = &quot;Due&quot;
*/
define &quot;Client is due for Hib vaccination&quot;:
  &quot;Client is due for Hib vaccination Case 1&quot;
    or &quot;Client is due for Hib vaccination Case 2&quot;
    or &quot;Client is due for Hib vaccination Case 3&quot;

/*
@output: Client is due for Hib vaccination Guidance
@guidance = Should vaccinate client with first Hib dose as no Hib doses were administered and client is within appropriate age range. 
Check for contraindications.
@guidance = Should vaccinate client with second Hib dose as client's age was less than 1 year when first Hib dose was administered and the latest Hib dose was administered more than 4 weeks ago. 
Check for contraindications.
@guidance = Should vaccinate client with third Hib dose as client is within appropriate age range and latest Hib dose was administered more than 4 weeks ago. 
Check for contraindications.
*/
define &quot;Client is due for Hib vaccination Guidance&quot;:
  case
    when &quot;Client is due for Hib vaccination Case 1&quot; then 'Should vaccinate client with first Hib dose as no Hib doses were administered and client is within appropriate age range. 
Check for contraindications.'
    when &quot;Client is due for Hib vaccination Case 2&quot; then 'Should vaccinate client with second Hib dose as client\'s age was less than 1 year when first Hib dose was administered and the latest Hib dose was administered more than 4 weeks ago. 
Check for contraindications.'
    when &quot;Client is due for Hib vaccination Case 3&quot; then 'Should vaccinate client with third Hib dose as client is within appropriate age range and latest Hib dose was administered more than 4 weeks ago. 
Check for contraindications.'
    else ''
  end

/*
@output: Hib immunization schedule is complete Case 1
@pseudocode: &quot;Immunization recommendation status&quot; = &quot;Complete&quot;
&quot;Completed the primary vaccination series&quot; = TRUE (where &quot;Vaccine type&quot; = &quot;Hib-containing vaccines&quot;)
*/
define &quot;Hib immunization schedule is complete Case 1&quot;:
  Encounter.&quot;One Hib primary series dose was administered&quot;

/*
@output: Hib immunization schedule is complete Case 2
@pseudocode: &quot;Immunization recommendation status&quot; = &quot;Complete&quot;

  Encounter.&quot;Three Hib primary series doses were administered&quot;

/*
@output: Hib immunization schedule is complete
@pseudocode: &quot;Immunization recommendation status&quot; = &quot;Complete&quot;
&quot;Completed the primary vaccination series&quot; = TRUE (where &quot;Vaccine type&quot; = &quot;Hib-containing vaccines&quot;)
*/
define &quot;Hib immunization schedule is complete&quot;:
  &quot;Hib immunization schedule is complete Case 1&quot;
    or &quot;Hib immunization schedule is complete Case 2&quot;

/*
@output: Hib immunization schedule is complete Guidance
@guidance = Hib immunization schedule is complete as client's age was more than 1 year when first Hib dose was administered. One Hib primary series dose was administered.
Check for any other vaccines due.
@guidance = Hib immunization schedule is complete. Three Hib primary series doses were administered. 
Check for any vaccines due.
*/
define &quot;Hib immunization schedule is complete Guidance&quot;:
  case
    when &quot;Hib immunization schedule is complete Case 1&quot; then 'Hib immunization schedule is complete as client\'s age was more than 1 year when first Hib dose was administered. One Hib primary series dose was administered.
Check for any other vaccines due.'
    when &quot;Hib immunization schedule is complete Case 2&quot; then 'Hib immunization schedule is complete. Three Hib primary series doses were administered. 
Check for any vaccines due.'
    else ''
  end

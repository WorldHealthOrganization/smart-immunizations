              <tr><td><pre><code class="language-cql">/*
 * Library: IMMZD18SPolioBirthDoseLogic (IMMZ.D18.S.Polio.Birth dose schedule)
 * Schedule Table: Birth dose schedule (in countries that are poliomyelitis (polio)-endemic countries or countries at high risk for importation and subsequent spread of polio that follow the combined bivalent oral polio vaccine [bOPV]–inactivated polio vaccine [IPV] schedule)
 */
library IMMZD18SPolioBirthDoseLogic

using FHIR version '4.0.1'
include FHIRHelpers version '4.0.1'

include WHOCommon called WC

include IMMZCommon called Common
include IMMZConcepts called Concepts

include IMMZEncounterElements called IE
include IMMZD2DTPolioEncounterElements called Encounter

parameter Today Date default Today()


context Patient


/*
@output: bOPV birth dose (a zero dose)
@description: Provision of bOPV birth dose (a zero dose)
@trigger: Child's birth
@pseudo: 
@triggerDate: &quot;Date of birth&quot;
*/
define &quot;bOPV birth dose (a zero dose)&quot;:
  not &quot;Polio birth dose (a zero dose) was administered&quot;
  and &quot;bOPV birth dose (a zero dose) Expiration&quot; after day of Today


/*
@output: bOPV birth dose (a zero dose) Create
@create: bOPV birth dose (a zero dose) should be provided within the first week of life and if the client is in a polio-endemic country or a country at high risk for importation and subsequent spread of polio
*/
define &quot;bOPV birth dose (a zero dose) Create&quot;:
  if &quot;bOPV birth dose (a zero dose)&quot;
  then 'bOPV birth dose (a zero dose) should be provided within the first week of life and if the client is in a polio-endemic country or a country at high risk for importation and subsequent spread of polio' + '
Due Date: ' + ToString(&quot;bOPV birth dose (a zero dose) Due Date&quot;) + '
Overdue: ' + ToString(&quot;bOPV birth dose (a zero dose) Overdue&quot;) + '
Expiration: ' + ToString(&quot;bOPV birth dose (a zero dose) Expiration&quot;)
  else ''


/*
@dynamicValue: bOPV birth dose (a zero dose) Due Date
@pseudocode: &quot;Date of birth&quot;
*/
define &quot;bOPV birth dose (a zero dose) Due Date&quot;:
  if &quot;bOPV birth dose (a zero dose)&quot; then Patient.birthDate
  else null


/*
@dynamicValue: bOPV birth dose (a zero dose) Overdue
@pseudocode: &quot;Date of birth&quot; + 1 day
*/
define &quot;bOPV birth dose (a zero dose) Overdue&quot;:
  if &quot;bOPV birth dose (a zero dose)&quot; then Patient.birthDate + 1 day
  else null


/*
@dynamicValue: bOPV birth dose (a zero dose) Expiration
@pseudocode: &quot;Date of birth&quot; + 1 week
*/
define &quot;bOPV birth dose (a zero dose) Expiration&quot;:
  Patient.birthDate + 1 week

  
/*
@complete: Polio birth dose (a zero dose) was administered
@pseudocode: Count of vaccines administered (where &quot;Vaccine type&quot; = &quot;Poliovirus-containing vaccines&quot; and &quot;Type of dose&quot; = &quot;Dose 0&quot; AND &quot;Birth dose&quot; = TRUE) = 1
*/
define &quot;Polio birth dose (a zero dose) was administered&quot;:
  Encounter.&quot;bOPV birth dose (a zero dose) was administered&quot;



/*
@test: Test expected results based on example patients
*/
define &quot;Test Validation&quot;:
  case
    when Patient.id = 'Polio08.2' then &quot;bOPV birth dose (a zero dose)&quot;
    when Patient.id = 'Polio09.2' then not &quot;bOPV birth dose (a zero dose)&quot;
    when Patient.id = 'Polio10.1' then not &quot;bOPV birth dose (a zero dose)&quot;
    else 'No test case set'
  end

@dynamicValue: Guidance

  case

    when &quot;Mumps vaccination could be contraindicated. Clinical judgement is required.&quot; then &quot;Mumps vaccination could be contraindicated. Clinical judgement is required. Guidance&quot;

*/

  Encounter.&quot;The client is severely immunosuppressed&quot;

/*
@output: Mumps vaccination is contraindicated

*/

  &quot;Mumps vaccination is contraindicated Case 1&quot;
    or &quot;Mumps vaccination is contraindicated Case 2&quot;
    or &quot;Mumps vaccination is contraindicated Case 3&quot;

/*
@output: Mumps vaccination is contraindicated Guidance
@guidance = Do not vaccinate client with mumps as mumps vaccination is contraindicated for pregnant individuals.
@guidance = Do not vaccinate client with mumps as mumps vaccination is contraindicated in individuals with immunodeficiency syndromes.
@guidance = Do not vaccinate client with mumps as mumps vaccination is contraindicated in immunosuppressed individuals.
*/
define &quot;Mumps vaccination is contraindicated Guidance&quot;:
  case
    when &quot;Mumps vaccination is contraindicated Case 1&quot; then 'Do not vaccinate client with mumps as mumps vaccination is contraindicated for pregnant individuals.'
    when &quot;Mumps vaccination is contraindicated Case 2&quot; then 'Do not vaccinate client with mumps as mumps vaccination is contraindicated in individuals with immunodeficiency syndromes.'
    when &quot;Mumps vaccination is contraindicated Case 3&quot; then 'Do not vaccinate client with mumps as mumps vaccination is contraindicated in immunosuppressed individuals.'
    else ''
  end

@output: Mumps vaccination could be contraindicated. Clinical judgement is required.

  Encounter.&quot;The client has allergy to vaccine components&quot;

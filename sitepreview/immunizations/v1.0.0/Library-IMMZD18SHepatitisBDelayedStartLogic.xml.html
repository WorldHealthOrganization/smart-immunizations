 * Library: IMMZD18SHepatitisBDelayedStartLogic (IMMZ.D18.S.Hepatitis B.Delayed start schedule)
 * Schedule Table: 3-dose schedule (no birth dose + 3 doses), delayed start
 */
library IMMZD18SHepatitisBDelayedStartLogic

using FHIR version '4.0.1'
include FHIRHelpers version '4.0.1'

include WHOCommon called WC

include IMMZCommon called Common
include IMMZConcepts called Concepts

include IMMZEncounterElements called IE
include IMMZD2DTHepatitisBEncounterElements called Encounter

parameter Today Date default Today()


context Patient


/*
@output: Hepatitis B dose 1
@description: Provision of hepatitis B dose 1
@trigger: Child's birth
@pseudo: 
@triggerDate: &quot;Date of birth&quot;
*/
define &quot;Hepatitis B dose 1&quot;:
  not &quot;First hepatitis B dose was administered&quot;
  and not &quot;Second hepatitis B dose was administered&quot;
  and not &quot;Third hepatitis B dose was administered. The primary series has been completed&quot;


/*
@output: Hepatitis B dose 1 Create
@create: If delayed or interrupted scheduling of vaccination for children, adolescents and adults, 3 doses are recommended, with the second dose administered at least 1 month after the first, and the third dose 6 months after the first dose.
*/
define &quot;Hepatitis B dose 1 Create&quot;:
  if &quot;Hepatitis B dose 1&quot;
  then 'If delayed or interrupted scheduling of vaccination for children, adolescents and adults, 3 doses are recommended, with the second dose administered at least 1 month after the first, and the third dose 6 months after the first dose.' + '
Due Date: ' + ToString(&quot;Hepatitis B dose 1 Due Date&quot;) + '
Overdue: ' + ToString(&quot;Hepatitis B dose 1 Overdue&quot;)
  else ''


/*
@dynamicValue: Hepatitis B dose 1 Due Date
@pseudocode: &quot;Date of birth&quot;
*/
define &quot;Hepatitis B dose 1 Due Date&quot;:
  if &quot;Hepatitis B dose 1&quot; then Patient.birthDate
  else null


/*
@dynamicValue: Hepatitis B dose 1 Overdue
@pseudocode: Today's date − &quot;Date of birth&quot; &gt; Member States defined lower limit
*/
define &quot;Hepatitis B dose 1 Overdue&quot;:
  Patient.birthDate + (Encounter.HepBLowerLimitWeeks * 1 weeks)


/*
@dynamicValue: Hepatitis B dose 1 Expiration
@pseudocode: To be determined by Member States; however, there is no recommended expiration date and individuals are always eligible to be vaccinated.
*/
define &quot;Hepatitis B dose 1 Expiration&quot;:
  null

  
/*
@complete: First hepatitis B dose was administered
@pseudocode: Count of vaccines administered (where &quot;Vaccine type&quot; = &quot;Hepatitis B-containing vaccines&quot;) = 1
*/
define &quot;First hepatitis B dose was administered&quot;:
  Encounter.&quot;One hepatitis B primary series dose was administered&quot;



/*
@output: Hepatitis B dose 2
@description: Provision of hepatitis B dose 2
@trigger: First hepatitis B dose was administered
@pseudo: Count of vaccines administered (where &quot;Vaccine type&quot; = &quot;Hepatitis B-containing vaccines&quot;) = 1
@triggerDate: Latest &quot;Date and time of vaccination&quot; (where &quot;Vaccine type&quot; = &quot;Hepatitis B-containing vaccines&quot;)
*/
define &quot;Hepatitis B dose 2&quot;:
  &quot;First hepatitis B dose was administered&quot;
  and not &quot;Second hepatitis B dose was administered&quot;
  and not &quot;Third hepatitis B dose was administered. The primary series has been completed&quot;


/*
@output: Hepatitis B dose 2 Create
@create: If delayed or interrupted scheduling of vaccination for children, adolescents and adults, 3 doses are recommended, with the second dose administered at least 1 month after the first, and the third dose 6 months after the first dose.
*/
define &quot;Hepatitis B dose 2 Create&quot;:
  if &quot;Hepatitis B dose 2&quot;
  then 'If delayed or interrupted scheduling of vaccination for children, adolescents and adults, 3 doses are recommended, with the second dose administered at least 1 month after the first, and the third dose 6 months after the first dose.' + '
Due Date: ' + ToString(&quot;Hepatitis B dose 2 Due Date&quot;)
  else ''


/*
@dynamicValue: Hepatitis B dose 2 Due Date
@pseudocode: Latest &quot;Date and time of vaccination&quot; (where &quot;Vaccine type&quot; = &quot;Hepatitis B-containing vaccines&quot;) + 1 month
*/
define &quot;Hepatitis B dose 2 Due Date&quot;:
  if &quot;Hepatitis B dose 2&quot; then Encounter.&quot;Date of Latest Hepatitis B Dose&quot; + 1 month
  else null


/*
@dynamicValue: Hepatitis B dose 2 Overdue
@pseudocode: To be determined by Member States; however, there is no recommended overdue date.
*/
define &quot;Hepatitis B dose 2 Overdue&quot;:
  null

@dynamicValue: Hepatitis B dose 2 Expiration

*/
define &quot;Hepatitis B dose 2 Expiration&quot;:
  null

  
/*
@complete: Second hepatitis B dose was administered
@pseudocode: Count of vaccines administered (where &quot;Vaccine type&quot; = &quot;Hepatitis B-containing vaccines&quot;) = 2
*/
define &quot;Second hepatitis B dose was administered&quot;:
  Encounter.&quot;Two hepatitis B primary series doses were administered&quot;



/*
@output: Hepatitis B dose 3
@description: Provision of hepatitis B dose 3
@trigger: Second hepatitis B dose was administered
@pseudo: Count of vaccines administered (where &quot;Vaccine type&quot; = &quot;Hepatitis B-containing vaccines&quot;) = 2
@triggerDate: &quot;Date and time of vaccination&quot; (where &quot;Vaccine type&quot; = &quot;Hepatitis B-containing vaccines&quot; and &quot;Dose number&quot; = 1)
*/
define &quot;Hepatitis B dose 3&quot;:
  &quot;Second hepatitis B dose was administered&quot;
  and not &quot;Third hepatitis B dose was administered. The primary series has been completed&quot;


/*
@output: Hepatitis B dose 3 Create
@create: If delayed or interrupted scheduling of vaccination for children, adolescents and adults, 3 doses are recommended, with the second dose administered at least 1 month after the first, and the third dose 6 months after the first dose.
*/
define &quot;Hepatitis B dose 3 Create&quot;:
  if &quot;Hepatitis B dose 3&quot;
  then 'If delayed or interrupted scheduling of vaccination for children, adolescents and adults, 3 doses are recommended, with the second dose administered at least 1 month after the first, and the third dose 6 months after the first dose.' + '
Due Date: ' + ToString(&quot;Hepatitis B dose 3 Due Date&quot;)
  else ''


/*
@dynamicValue: Hepatitis B dose 3 Due Date
@pseudocode: &quot;Date and time of vaccination&quot; (where &quot;Vaccine type&quot; = &quot;Hepatitis B-containing vaccines&quot; and &quot;Dose number&quot; = 1) + 6 months
*/
define &quot;Hepatitis B dose 3 Due Date&quot;:
  if &quot;Hepatitis B dose 3&quot; then Encounter.&quot;Date of First Hepatitis B Dose&quot; + 6 months
  else null


/*
@dynamicValue: Hepatitis B dose 3 Overdue

  null

  
/*
@complete: Third hepatitis B dose was administered. The primary series has been completed

*/
define &quot;Third hepatitis B dose was administered. The primary series has been completed&quot;:
  Encounter.&quot;Three hepatitis B primary series doses were administered&quot;

define &quot;Test Validation&quot;:

    when Patient.id = 'HepatitisB54.3' then &quot;Hepatitis B dose 3&quot;

    when Patient.id = 'HepatitisB56.1' then not &quot;Hepatitis B dose 3&quot; and &quot;Third hepatitis B dose was administered. The primary series has been completed&quot;

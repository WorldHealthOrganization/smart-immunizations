using FHIR version '4.0.1'

include WHOConcepts

include WHOElements called WE

include IMMZCommon called Common
include IMMZConcepts called Concepts
include IMMZEncounterElements called Encounter

include IMMZD2DTMalariaElements called MalariaElements

/*
@internal: Malaria containing Doses Administered to Patient

  Encounter.&quot;Doses Administered to Patient&quot; I

    I.vaccineCode in Concepts.&quot;Malaria vaccines&quot;

/*
@internal: Malaria containing Doses Administered to Patient that are in the Primary series

  Encounter.&quot;Current Patient Age In Months&quot; &lt; 5

@input: No malaria primary series doses were administered

@code: Client's age is more than or equal to 5 months-41

  &quot;Number of Malaria Primary Series Doses Administered&quot; = 1

/*
 * Library: IMMZD18STyphoidVIPSLogic (IMMZ.D18.S.Typhoid.ViPS schedule)
 * Schedule Table: Unconjugated Vi polysaccharide (ViPS), 1 dose schedule
 */
library IMMZD18STyphoidVIPSLogic

using FHIR version '4.0.1'
include FHIRHelpers version '4.0.1'

include WHOCommon called WC

include IMMZCommon called Common
include IMMZConcepts called Concepts

include IMMZEncounterElements called IE
include IMMZD2DTTyphoidEncounterElements called Encounter

parameter Today Date default Today()

context Patient


/*
@output: Typhoid dose 1
@description: Provision of the typhoid ViPS dose 1
@trigger: Child's birth
@triggerDate: &quot;Date of birth&quot;
*/
define &quot;Typhoid dose 1&quot;:
  Encounter.&quot;No typhoid primary series doses were administered&quot;
  and not &quot;One typhoid dose from the primary series was administered. The primary series has been completed&quot;

/*
@output: Typhoid dose 1 Create
@create: Typhoid dose 1 should be provided if the client's age is over 2 years, and the region recommends typhoid vaccination.
*/
define &quot;Typhoid dose 1 Create&quot;:
  if &quot;Typhoid dose 1&quot; 
  then 'Typhoid dose 1 should be provided if the client\'s age is over 2 years, and the region recommends typhoid vaccination.' + '
Due Date: ' + ToString(&quot;Typhoid dose 1 Due Date&quot;)
  else ''

/*
@dynamicValue: Typhoid dose 1 Due Date
@pseudocode: &quot;Date of birth&quot; + 2 years
*/
define &quot;Typhoid dose 1 Due Date&quot;:
  if &quot;Typhoid dose 1&quot; then Patient.birthDate + 2 years
  else null

/*
@dynamicValue: Typhoid dose 1 Overdue
@pseudocode: To be determined by Member States; however, there is no recommended overdue date and individuals are always eligible to be vaccinated.
*/
define &quot;Typhoid dose 1 Overdue&quot;:
  null

/*
@dynamicValue: Typhoid dose 1 Expiration
@pseudocode: To be determined by Member States; however, there is no recommended expiration date and individuals are always eligible to be vaccinated.
*/
define &quot;Typhoid dose 1 Expiration&quot;:
  null
  
/*
@complete: One typhoid dose from the primary series was administered. The primary series has been completed
@pseudocode: &quot;Completed the primary vaccination series&quot; = TRUE (where &quot;Vaccine type&quot; = &quot;Typhoid vaccines&quot;)
*/
define &quot;One typhoid dose from the primary series was administered. The primary series has been completed&quot;:
  Encounter.&quot;One typhoid primary series dose was administered&quot;

/*
@output: Typhoid booster dose(s)
@description: Provision of the typhoid ViPS booster dose
@trigger: The primary series has been completed
&quot;Completed the primary vaccination series&quot; = TRUE (where &quot;Vaccine type&quot; = &quot;Typhoid vaccines&quot;)
@triggerDate: Latest &quot;Date and time of vaccination&quot; (where &quot;Vaccine type&quot; = &quot;Typhoid vaccines&quot;)
*/
define &quot;Typhoid booster dose(s)&quot;:
  &quot;One typhoid dose from the primary series was administered. The primary series has been completed&quot;

/*
@output: Typhoid booster dose(s) Create
@create: Revaccination is recommended every 3 years for ViPS.
*/
define &quot;Typhoid booster dose(s) Create&quot;:
  if &quot;Typhoid booster dose(s)&quot; 
  then 'Revaccination is recommended every 3 years for ViPS.' + '
Due Date: ' + ToString(&quot;Typhoid booster dose(s) Due Date&quot;)
  else ''

/*
@dynamicValue: Typhoid booster dose(s) Due Date
@pseudocode: Latest &quot;Date and time of vaccination&quot; (where &quot;Vaccine type&quot; = &quot;Typhoid vaccines&quot;) + 3 years
*/
define &quot;Typhoid booster dose(s) Due Date&quot;:
  if &quot;Typhoid booster dose(s)&quot; then Encounter.&quot;Date of Latest Typhoid Dose&quot; + 3 years
  else null

/*
@dynamicValue: Typhoid booster dose(s) Overdue
@pseudocode: To be determined by Member States; however, there is no recommended overdue date and individuals are always eligible to be vaccinated.
*/
define &quot;Typhoid booster dose(s) Overdue&quot;:
  null

/*
@dynamicValue: Typhoid booster dose(s) Expiration
@pseudocode: To be determined by Member States; however, there is no recommended expiration date and individuals are always eligible to be vaccinated.
*/
define &quot;Typhoid booster dose(s) Expiration&quot;:
  null
  
/*
@complete: Typhoid booster dose was administered less than 3 years ago
@pseudocode: Count of vaccines administered (where &quot;Vaccine type&quot; = &quot;Typhoid vaccines&quot; and &quot;Type of dose&quot; = &quot;Booster dose&quot;)  ≥ 1 AND Today's date - Latest &quot;Date and time of vaccination&quot; (where &quot;Vaccine type&quot; = &quot;Typhoid vaccines&quot;) &lt; 3 years
*/
define &quot;Typhoid booster dose was administered less than 3 years ago&quot;:
  Encounter.&quot;Date of Latest Typhoid Dose&quot; is not null
  and duration in years between Encounter.&quot;Date of Latest Typhoid Dose&quot; and Today &lt; 3

/*
@test: Test expected results based on example patients
*/
define &quot;Test Validation&quot;:
  case
    when Patient.id = '19.2' then &quot;Typhoid dose 1&quot;
    when Patient.id = '20.12' then &quot;Typhoid dose 1&quot;
    when Patient.id = '21.13' then &quot;Typhoid booster dose(s)&quot;
    when Patient.id = '22.13' then &quot;Typhoid booster dose(s)&quot;
    else 'No test case set'
  end

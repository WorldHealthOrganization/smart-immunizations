              <tr><td><pre><code class="language-cql">/*
 * Library: IMMZD2DTJELiveRecombinantVaccineLogic (IMMZ.D2.DT.JE.Live recombinant vaccine)
 * Rule: Determine if the client is due for a Japanese encephalitis (JE) vaccination according to the national immunization schedule
 * Decision Table: Live recombinant vaccination schedule (1-dose scheme)
JE vaccination should be integrated into national immunization schedules in all areas where JE is recognized as a public health priority
 * Trigger: IMMZ.D2 Determine required vaccination(s) if any
 */
library IMMZD2DTJELiveRecombinantVaccineLogic

using FHIR version '4.0.1'
include FHIRHelpers version '4.0.1'

include IMMZD2DTJEEncounterElements called Encounter

parameter Today Date default Today()

context Patient


/*
@dynamicValue: Guidance
*/
define &quot;Guidance&quot;:
  case 
    when &quot;Client is not due for JE vaccination&quot; then &quot;Client is not due for JE vaccination Guidance&quot;
    when &quot;Client is due for JE vaccination&quot; then &quot;Client is due for JE vaccination Guidance&quot;
    when &quot;JE immunization schedule is complete&quot; then &quot;JE immunization schedule is complete Guidance&quot; 
    else ''
  end
  
define &quot;Has Guidance&quot;:
  &quot;Guidance&quot; is not null and &quot;Guidance&quot; != ''

/*
@output: Client is not due for JE vaccination Case 1
@pseudocode: &quot;Immunization recommendation status&quot; = &quot;Not due&quot;
*/
define &quot;Client is not due for JE vaccination Case 1&quot;:
  Encounter.&quot;Client's age is less than 9 months&quot;

/*
@output: Client is not due for JE vaccination Case 2
@pseudocode: &quot;Immunization recommendation status&quot; = &quot;Not due&quot;
*/
define &quot;Client is not due for JE vaccination Case 2&quot;:
  Encounter.&quot;No JE primary series doses were administered&quot;
    and Encounter.&quot;Client's age is more than or equal to 9 months&quot;
    and Encounter.&quot;Live vaccine was administered in the last 4 weeks&quot;

/*
@output: Client is not due for JE vaccination
@pseudocode: &quot;Immunization recommendation status&quot; = &quot;Not due&quot;
*/
define &quot;Client is not due for JE vaccination&quot;:
  &quot;Client is not due for JE vaccination Case 1&quot;
    or &quot;Client is not due for JE vaccination Case 2&quot;

/*
@output: Client is not due for JE vaccination Guidance
@guidance = Should not vaccinate client with JE dose as client's age is less than 9 months. 
Check for any vaccines due and inform the caregiver of when to come back for the first JE dose.
@guidance = Should not vaccinate client with JE dose as live vaccine was administered in the last 4 weeks. 
Check for any other vaccines due and inform the caregiver of when to come back for the first JE dose.
*/
define &quot;Client is not due for JE vaccination Guidance&quot;:
  case
    when &quot;Client is not due for JE vaccination Case 1&quot; then 'Should not vaccinate client with JE dose as client\'s age is less than 9 months. 
Check for any vaccines due and inform the caregiver of when to come back for the first JE dose.'
    when &quot;Client is not due for JE vaccination Case 2&quot; then 'Should not vaccinate client with JE dose as live vaccine was administered in the last 4 weeks. 
Check for any other vaccines due and inform the caregiver of when to come back for the first JE dose.'
    else ''
  end

/*
@output: Client is due for JE vaccination
@pseudocode: &quot;Immunization recommendation status&quot; = &quot;Due&quot;
*/
define &quot;Client is due for JE vaccination&quot;:
  Encounter.&quot;No JE primary series doses were administered&quot;
    and Encounter.&quot;Client's age is more than or equal to 9 months&quot;
    and Encounter.&quot;No live vaccine was administered in the last 4 weeks&quot;

/*
@output: Client is due for JE vaccination Guidance
@guidance: Should vaccinate client with JE dose as no JE dose was administered, client is within the appropriate age range and no live vaccine was administered in past 4 weeks. 
Check for contraindications.
*/
define &quot;Client is due for JE vaccination Guidance&quot;:
  'Should vaccinate client with JE dose as no JE dose was administered, client is within the appropriate age range and no live vaccine was administered in past 4 weeks. 
Check for contraindications.'

/*
@output: JE immunization schedule is complete
@pseudocode: &quot;Immunization recommendation status&quot; = &quot;Complete&quot;
&quot;Completed the primary vaccination series&quot; = TRUE (where &quot;Vaccine type&quot; = &quot;JE vaccines&quot;)
*/
define &quot;JE immunization schedule is complete&quot;:
  Encounter.&quot;One JE primary series doses were administered&quot;

/*
@output: JE immunization schedule is complete Guidance
@guidance: JE immunization schedule is complete. One JE primary series dose was administered. 
Check for any other vaccines due.
*/
define &quot;JE immunization schedule is complete Guidance&quot;:
  'JE immunization schedule is complete. One JE primary series dose was administered. 
Check for any other vaccines due.'


/*
@test: Test expected results based on example patients
*/
define &quot;Test Validation&quot;:
  case
    when Patient.id = 'JE31.1' then &quot;Client is not due for JE vaccination Case 1&quot; and &quot;Guidance&quot; = 'Should not vaccinate client with JE dose as client\'s age is less than 9 months. 
Check for any vaccines due and inform the caregiver of when to come back for the first JE dose.'
    when Patient.id = 'JE32.3' then &quot;Client is due for JE vaccination&quot; and &quot;Guidance&quot; = 'Should vaccinate client with JE dose as no JE dose was administered, client is within the appropriate age range and no live vaccine was administered in past 4 weeks. 
Check for contraindications.'
    when Patient.id = 'JE33.3' then &quot;Client is not due for JE vaccination Case 2&quot; and &quot;Guidance&quot; = 'Should not vaccinate client with JE dose as live vaccine was administered in the last 4 weeks. 
Check for any other vaccines due and inform the caregiver of when to come back for the first JE dose.'
    when Patient.id = 'JE34.1' then &quot;JE immunization schedule is complete&quot; and &quot;Guidance&quot; = 'JE immunization schedule is complete. One JE primary series dose was administered. 
Check for any other vaccines due.'
    else 'No test case set'
  end

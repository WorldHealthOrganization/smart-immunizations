/*
 * Library: IMMZD5DTBCGCILogic (IMMZ.D5.DT.BCG contraindications)
 * Rule: Check for contraindications before administering the vaccine(s) due	
 * Decision Table: Potential contraindications
 * Trigger: IMMZ.D5 Determine vaccine(s) to be administered based on contraindications
 */
library IMMZD5DTBCGCILogic

using FHIR version '4.0.1'
include FHIRHelpers version '4.0.1'

include IMMZD5DTBCGEncounterElements called Encounter

parameter Today Date default Today()

context Patient


/*
@dynamicValue: Draft Medication Request ID for BCG dose
*/
define &quot;Draft Medication Request ID for BCG dose&quot;:
	First(Encounter.&quot;Draft Medication Request for BCG dose&quot;).id

/*
@dynamicValue: Guidance
*/
define &quot;Guidance&quot;:
  case
    when &quot;Bacille Calmette–Guérin (BCG) vaccination is contraindicated&quot; then &quot;Bacille Calmette–Guérin (BCG) vaccination is contraindicated Guidance&quot;
    when &quot;BCG vaccination could be contraindicated. Clinical judgement is required. Create a clinical note.&quot; then &quot;BCG vaccination could be contraindicated. Clinical judgement is required. Create a clinical note. Guidance&quot;
    when &quot;BCG vaccination is contraindicated&quot; then &quot;BCG vaccination is contraindicated Guidance&quot;
    else ''
  end

define &quot;Has Guidance&quot;:
  &quot;Guidance&quot; is not null and &quot;Guidance&quot; != ''

/*
@output: Bacille Calmette–Guérin (BCG) vaccination is contraindicated
@pseudocode: '&quot;Immunization recommendation status&quot; = &quot;Contraindicated&quot;
*/
define &quot;Bacille Calmette–Guérin (BCG) vaccination is contraindicated&quot;:
  Encounter.&quot;The client is pregnant&quot;

/*
@output: Bacille Calmette–Guérin (BCG) vaccination is contraindicated Guidance
@guidance: Do not vaccinate client with BCG as BCG vaccination is not recommended during pregnancy
*/

  'Do not vaccinate client with BCG as BCG vaccination is not recommended during pregnancy'

*/

  Encounter.&quot;The client is exposed to immunosuppressive treatment&quot;
    and Encounter.&quot;Client's age is more than or equal to 1 year&quot;

/*
@output: BCG vaccination could be contraindicated. Clinical judgement is required. Create a clinical note.

*/
define &quot;BCG vaccination could be contraindicated. Clinical judgement is required. Create a clinical note.&quot;:
  &quot;BCG vaccination could be contraindicated. Clinical judgement is required. Create a clinical note. Case 1&quot;
    or &quot;BCG vaccination could be contraindicated. Clinical judgement is required. Create a clinical note. Case 2&quot;

/*
@output: BCG vaccination could be contraindicated. Clinical judgement is required. Create a clinical note. Guidance
@guidance: Do not vaccinate client with BCG if the client had previous allergic reaction to any component of the vaccine
@guidance: Do not vaccinate client with BCG if client is exposed to or receives immunosuppressive treatment
*/
define &quot;BCG vaccination could be contraindicated. Clinical judgement is required. Create a clinical note. Guidance&quot;:
  case
    when &quot;BCG vaccination could be contraindicated. Clinical judgement is required. Create a clinical note. Case 1&quot; then 'Do not vaccinate client with BCG if the client had previous allergic reaction to any component of the vaccine'
    when &quot;BCG vaccination could be contraindicated. Clinical judgement is required. Create a clinical note. Case 2&quot; then 'Do not vaccinate client with BCG if client is exposed to or receives immunosuppressive treatment'
    else ''
  end

@output: BCG vaccination is contraindicated Case 1
@pseudocode: '&quot;Immunization recommendation status&quot; = &quot;Contraindicated&quot;
*/
define &quot;BCG vaccination is contraindicated Case 1&quot;:
  Encounter.&quot;The client has immunodeficiency syndromes&quot;

*/
define &quot;BCG vaccination is contraindicated Case 2&quot;:
  Encounter.&quot;The client is exposed to immunosuppressive treatment&quot;
    and Encounter.&quot;Client's age is less than 1 year&quot;

/*
@output: BCG vaccination is contraindicated

*/
define &quot;BCG vaccination is contraindicated&quot;:
  &quot;BCG vaccination is contraindicated Case 1&quot;

@output: BCG vaccination is contraindicated Guidance
@guidance: Do not vaccinate client with BCG as BCG vaccination is contraindicated for clients with immunodeficiency syndromes
@guidance: Do not vaccinate client with BCG as BCG vaccination is contraindicated for clients undergoing immunosuppressive treatment
*/

  case

    when &quot;BCG vaccination is contraindicated Case 2&quot; then 'Do not vaccinate client with BCG as BCG vaccination is contraindicated for clients undergoing immunosuppressive treatment'

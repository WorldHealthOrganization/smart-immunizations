/*
 * Library: IMMZD18STyphoidTy21aLogic (IMMZ.D18.S.Typhoid.Ty21a schedule)
 * Schedule Table: Live attenuated Ty21a vaccine, 3-dose schedule
 */
library IMMZD18STyphoidTy21aLogic

using FHIR version '4.0.1'
include FHIRHelpers version '4.0.1'

include WHOCommon called WC

include IMMZCommon called Common
include IMMZConcepts called Concepts

include IMMZEncounterElements called IE
include IMMZD2DTTyphoidEncounterElements called Encounter

parameter Today Date default Today()

context Patient


/*
@output: Typhoid Ty21a primary series
@description: Provision of the typhoid Ty21a, 3-dose oral primary series immunization
@trigger: Child's birth
@triggerDate: &quot;Date of birth&quot;
*/
define &quot;Typhoid Ty21a primary series&quot;:
  Encounter.&quot;No typhoid primary series doses were administered&quot;
  and not &quot;The primary series has been completed&quot;

/*
@output: Typhoid Ty21a primary series Create
@create: Ty21a – 3-doses to be administered orally every second day from 6 years of age.
*/
define &quot;Typhoid Ty21a primary series Create&quot;:
  if &quot;Typhoid Ty21a primary series&quot; 
  then 'Ty21a – 3-doses to be administered orally every second day from 6 years of age.' + '
Due Date: ' + ToString(&quot;Typhoid Ty21a primary series Due Date&quot;)
  else ''

/*
@dynamicValue: Typhoid Ty21a primary series Due Date
@pseudocode: &quot;Date of birth&quot; + 6 years
*/
define &quot;Typhoid Ty21a primary series Due Date&quot;:
  if &quot;Typhoid Ty21a primary series&quot; then Patient.birthDate + 6 years
  else null

/*
@dynamicValue: Typhoid Ty21a primary series Overdue
@pseudocode: To be determined by Member States; however, there is no recommended overdue date and individuals are always eligible to be vaccinated.
*/
define &quot;Typhoid Ty21a primary series Overdue&quot;:
  null

/*
@dynamicValue: Typhoid Ty21a primary series Expiration
@pseudocode: To be determined by Member States; however, there is no recommended expiration date and individuals are always eligible to be vaccinated.
*/
define &quot;Typhoid Ty21a primary series Expiration&quot;:
  null
  
/*
@complete: The primary series has been completed
@pseudocode: &quot;Completed the primary vaccination series&quot; = TRUE (where &quot;Vaccine type&quot; = &quot;Typhoid vaccines&quot;)
*/
define &quot;The primary series has been completed&quot;:
  Encounter.&quot;Primary series is complete&quot;

/*
@output: Typhoid booster dose(s): repeat 3-doses series
@description: Provision of the typhoid booster dose(s): revaccination
@trigger: The primary series has been completed
&quot;Completed the primary vaccination series&quot; = TRUE (where &quot;Vaccine type&quot; = &quot;Typhoid vaccines&quot;)
@triggerDate: Latest &quot;Date and time of vaccination&quot; (where &quot;Vaccine type&quot; = &quot;Typhoid vaccines&quot;)
*/
define &quot;Typhoid booster dose(s) repeat 3-doses series&quot;:
  &quot;The primary series has been completed&quot;

/*
@output: Typhoid booster dose(s): repeat 3-doses series Create
@create: Revaccination (3-doses) is recommended every 3–7 years for Ty21a.
*/
define &quot;Typhoid booster dose(s) repeat 3-doses series Create&quot;:
  if &quot;Typhoid booster dose(s) repeat 3-doses series&quot; 
  then 'Revaccination (3-doses) is recommended every 3–7 years for Ty21a.' + '
Due Date: ' + ToString(&quot;Typhoid booster dose(s) repeat 3-doses series Due Date&quot;) + '
Overdue: ' + ToString(&quot;Typhoid booster dose(s) repeat 3-doses series Overdue&quot;)
  else ''

/*
@dynamicValue: Typhoid booster dose(s): repeat 3-doses series Due Date
@pseudocode: Latest &quot;Date and time of vaccination&quot; (where &quot;Vaccine type&quot; = &quot;Typhoid vaccines&quot;) + 3 years
*/
define &quot;Typhoid booster dose(s) repeat 3-doses series Due Date&quot;:
  if &quot;Typhoid booster dose(s) repeat 3-doses series&quot; then Encounter.&quot;Date of Latest Typhoid Dose&quot; + 3 years
  else null

/*
@dynamicValue: Typhoid booster dose(s): repeat 3-doses series Overdue
@pseudocode: Latest &quot;Date and time of vaccination&quot; (where &quot;Vaccine type&quot; = &quot;Typhoid vaccines&quot;) + 7 years
*/
define &quot;Typhoid booster dose(s) repeat 3-doses series Overdue&quot;:
  if &quot;Typhoid booster dose(s) repeat 3-doses series&quot; then Encounter.&quot;Date of Latest Typhoid Dose&quot; + 7 years
  else null

/*
@dynamicValue: Typhoid booster dose(s): repeat 3-doses series Expiration
@pseudocode: To be determined by Member States; however, there is no recommended expiration date and individuals are always eligible to be vaccinated.
*/
define &quot;Typhoid booster dose(s) repeat 3-doses series Expiration&quot;:
  null
  
/*
@complete: The booster series has been completed
@pseudocode: &quot;Completed the booster series&quot; = TRUE (where &quot;Vaccine type&quot; = &quot;Typhoid vaccines&quot;)
*/
define &quot;The booster series has been completed&quot;:
  Encounter.&quot;Date of Latest Typhoid Dose&quot; is not null
  and duration in years between Encounter.&quot;Date of Latest Typhoid Dose&quot; and Today &lt; 3

/*
@test: Test expected results based on example patients

    when Patient.id = '34.1234' then &quot;Typhoid booster dose(s) repeat 3-doses series&quot;
    when Patient.id = '35.1234' then &quot;Typhoid booster dose(s) repeat 3-doses series&quot;
    else 'No test case set'
  end

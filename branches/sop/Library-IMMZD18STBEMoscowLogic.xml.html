include WHOCommon called WC

@trigger: Child's birth
@triggerDate: &quot;Date of birth&quot;

define &quot;Tick-borne encephalitis (TBE)-Moscow dose 1&quot;:
  Encounter.&quot;No TBE primary series doses were administered&quot;
  and not &quot;First TBE dose from the primary series was administered&quot;

@output: Tick-borne encephalitis (TBE)-Moscow dose 1 Create
@create: The TBE-Moscow vaccine is not licensed for use in children aged under 3 years.
*/
define &quot;Tick-borne encephalitis (TBE)-Moscow dose 1 Create&quot;:
  if &quot;Tick-borne encephalitis (TBE)-Moscow dose 1&quot; 
  then 'The TBE-Moscow vaccine is not licensed for use in children aged under 3 years.' + '
Due Date: ' + ToString(&quot;Tick-borne encephalitis (TBE)-Moscow dose 1 Due Date&quot;)
  else ''

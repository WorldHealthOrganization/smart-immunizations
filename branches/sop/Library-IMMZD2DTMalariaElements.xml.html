library IMMZD2DTMalariaElements

using FHIR version '4.0.1'

include WHOElements called WE

include IMMZConcepts called Concepts

context Patient

/*
@internal: Malaria containing Doses Administered to Patient
*/
define &quot;Malaria Doses Administered to Patient&quot;:
  Elements.&quot;Doses Administered to Patient&quot; I
  where
    I.vaccineCode in Concepts.&quot;Malaria vaccines&quot;

/*
@internal: Malaria containing Doses Administered to Patient that are in the Primary series
*/
define &quot;Malaria Primary Series Doses Administered to Patient&quot;:
  &quot;Malaria Doses Administered to Patient&quot;.seriesPrimary()

/*
@internal: Number of Malaria Primary Series doses
*/

  Count(&quot;Malaria Primary Series Doses Administered to Patient&quot;)

/*
@internal: Date and time of last Malaria dose

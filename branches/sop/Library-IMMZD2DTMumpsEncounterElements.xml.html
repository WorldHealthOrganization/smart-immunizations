library IMMZD2DTMumpsEncounterElements

using FHIR version '4.0.1'
include FHIRHelpers version '4.0.1'

include WHOConcepts
include WHOCommon called WC
include WHOElements called WE

include IMMZCommon called Common
include IMMZConcepts called Concepts

include IMMZEncounterElements called Encounter

include IMMZD2DTMumpsElements called MumpsElements

parameter Today Date default Today()
parameter EncounterId String

context Patient

/*
@internal: Mumps containing Doses Administered to Patient
*/
define &quot;Mumps Doses Administered to Patient&quot;:
  Encounter.&quot;Doses Administered to Patient&quot; I
  where
    I.vaccineCode in Concepts.&quot;Mumps-containing vaccines&quot;

/*
@internal: Mumps containing Doses Administered to Patient that are in the Primary series
*/
define &quot;Mumps Primary Series Doses Administered to Patient&quot;:
  &quot;Mumps Doses Administered to Patient&quot;.seriesPrimary()

/*
@internal: Number of Mumps Primary Series doses
*/
define &quot;Number of Mumps Primary Series Doses Administered&quot;:
  Count(&quot;Mumps Primary Series Doses Administered to Patient&quot;)

/*
@internal: Date and time of last Mumps dose
*/
define &quot;Date of Latest Mumps Dose&quot;:
  date from start of &quot;Mumps Doses Administered to Patient&quot;.mostRecent().occurrence.toInterval()

/*
@input: No mumps primary series doses were administered
@pseudocode: Count of vaccines administered (where &quot;Vaccine type&quot; = &quot;Mumps-containing vaccines&quot; and &quot;Type of dose&quot; = &quot;Primary series&quot;) = 0
@decision: 2-dose schedule (countries with mumps in schedule and low transmission of measles)									
*/
define &quot;No mumps primary series doses were administered&quot;:
  &quot;Number of Mumps Primary Series Doses Administered&quot; = 0

/*
@input: One mumps primary series dose was administered
@pseudocode: Count of vaccines administered (where &quot;Vaccine type&quot; = &quot;Mumps-containing vaccines&quot; and &quot;Type of dose&quot; = &quot;Primary series&quot;) = 1
@decision: 2-dose schedule (countries with mumps in schedule and low transmission of measles)									
*/
define &quot;One mumps primary series dose was administered&quot;:
  &quot;Number of Mumps Primary Series Doses Administered&quot; = 1

/*
@input: Two mumps primary series doses were administered
@pseudocode: Count of vaccines administered (where &quot;Vaccine type&quot; = &quot;Mumps-containing vaccines&quot; and &quot;Type of dose&quot; = &quot;Primary series&quot;) = 2
@decision: 2-dose schedule (countries with mumps in schedule and low transmission of measles)									
*/
define &quot;Two mumps primary series doses were administered&quot;:
  &quot;Number of Mumps Primary Series Doses Administered&quot; = 2

/*
@input: Client's age is less than 12 months
@pseudocode: Today's date − &quot;Date of birth&quot; &lt; 12 months
@decision: 2-dose schedule (countries with mumps in schedule and low transmission of measles)									
*/
define &quot;Client's age is less than 12 months&quot;:
  Encounter.&quot;Current Patient Age In Months&quot; &lt; 12

/*
@input: Client's age is more than or equal to 12 months
@pseudocode: Today's date − &quot;Date of birth&quot; ≥ 12 months
@decision: 2-dose schedule (countries with mumps in schedule and low transmission of measles)									
*/
define &quot;Client's age is more than or equal to 12 months&quot;:
  Encounter.&quot;Current Patient Age In Months&quot; &gt;= 12

/*
@input: The latest mumps dose was administered less than 4 weeks ago
@pseudocode: Today's date − &quot;Date and time of vaccination&quot; (where &quot;Vaccine type&quot; = &quot;Mumps-containing vaccines&quot;) &lt; 4 weeks
@decision: 2-dose schedule (countries with mumps in schedule and low transmission of measles)									
*/
define &quot;The latest mumps dose was administered less than 4 weeks ago&quot;:
  &quot;Date of Latest Mumps Dose&quot; is not null 
  and duration in weeks between &quot;Date of Latest Mumps Dose&quot; and Today &lt; 4

/*
@input: The latest mumps dose was administered more than 4 weeks ago
@pseudocode: Today's date − &quot;Date and time of vaccination&quot; (where &quot;Vaccine type&quot; = &quot;Mumps-containing vaccines&quot;) ≥ 4 weeks
@decision: 2-dose schedule (countries with mumps in schedule and low transmission of measles)									
*/
define &quot;The latest mumps dose was administered more than 4 weeks ago&quot;:
  not &quot;The latest mumps dose was administered less than 4 weeks ago&quot;

/*
@input: Live vaccine was administered in the past 4 weeks
@pseudocode: Today's date − latest &quot;Date and time of vaccination&quot; (where &quot;Live vaccine&quot; = TRUE) &lt; 4 weeks

*/
define &quot;No live vaccine was administered in the past 4 weeks&quot;:

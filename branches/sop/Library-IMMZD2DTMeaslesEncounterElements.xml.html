library IMMZD2DTMeaslesEncounterElements

using FHIR version '4.0.1'
include FHIRHelpers version '4.0.1'

include WHOConcepts
include WHOCommon called WC
include WHOElements called WE

include IMMZCommon called Common
include IMMZConcepts called Concepts

include IMMZEncounterElements called Encounter

include IMMZD2DTMeaslesElements called MeaslesElements

parameter Today Date default Today()
parameter EncounterId String

context Patient

/*
@internal: MCV containing Doses Administered to Patient
*/
define &quot;MCV Doses Administered to Patient&quot;:
  MeaslesElements.&quot;MCV Doses Administered to Patient&quot; I
  where
    I.occurrence.toInterval() on or before Today

/*
@internal: MCV containing Doses Administered to Patient that are in the Primary series
*/
define &quot;MCV Primary Series Doses Administered to Patient&quot;:
  MeaslesElements.&quot;MCV Primary Series Doses Administered to Patient&quot; I
  where
    I.occurrence.toInterval() on or before Today

/*
@internal: MCV containing Doses Administered to Patient that are in the Dose 0

/*
@internal: MCV containing Doses Administered to Patient that are in the Supplementary dose
*/
define &quot;MCV Supplementary Doses Administered to Patient&quot;:
  MeaslesElements.&quot;MCV Supplementary Doses Administered to Patient&quot; I
  where
    I.occurrence.toInterval() on or before Today

/*
@internal: Number of MCV Primary Series doses
*/
define &quot;Number of MCV Primary Series Doses Administered&quot;:
  Count(&quot;MCV Primary Series Doses Administered to Patient&quot;)

/*
@internal: Number of MCV Dose 0 doses
*/
define &quot;Number of MCV Dose 0 Doses Administered&quot;:
  Count(&quot;MCV Dose 0 Doses Administered to Patient&quot;)

*/

  Count(&quot;MCV Supplementary Doses Administered to Patient&quot;)

*/
define &quot;No measles primary series doses were administered&quot;:
  &quot;Number of MCV Primary Series Doses Administered&quot; = 0

/*
@input: MCV1 was administered
@pseudocode: Count of vaccines administered (where &quot;Vaccine type&quot; = &quot;Measles-containing vaccines&quot; and &quot;Type of dose&quot; = &quot;Primary series&quot;) = 1
*/
define &quot;MCV1 was administered&quot;:
  &quot;Number of MCV Primary Series Doses Administered&quot; = 1

/*
@input: MCV2 was administered
@pseudocode: Count of vaccines administered (where &quot;Vaccine type&quot; = &quot;Measles-containing vaccines&quot; and &quot;Type of dose&quot; = &quot;Primary series&quot;) = 2
*/
define &quot;MCV2 was administered&quot;:
  &quot;Number of MCV Primary Series Doses Administered&quot; = 2

@input: MCV0 was not administered
@pseudocode: Count of vaccines administered (where &quot;Vaccine type&quot; = &quot;Measles-containing vaccines
&quot; and &quot;Type of dose&quot; = &quot;Dose 0&quot;) = 0
*/
define &quot;MCV0 was not administered&quot;:
  &quot;Number of MCV Dose 0 Doses Administered&quot; = 0

/*
@input: MCV0 was administered
@pseudocode: Count of vaccines administered (where &quot;Vaccine type&quot; = &quot;Measles-containing vaccines
&quot; and &quot;Type of dose&quot; = &quot;Dose 0&quot;) = 1

define &quot;MCV0 was administered&quot;:
  &quot;Number of MCV Dose 0 Doses Administered&quot; = 1

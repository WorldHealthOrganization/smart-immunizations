include FHIRHelpers version '4.0.1'

include WHOConcepts
include WHOCommon called WC
include WHOElements called WE

include IMMZCommon called Common
include IMMZConcepts called Concepts
include IMMZElements called Elements

context Patient

/*
@internal: Draft Medication Request for Seasonal influenza dose
*/
define &quot;Draft Medication Request for Seasonal influenza dose&quot;:
  Elements.&quot;Draft Medication Request for Patient&quot; MR 
    where MR.medication in Concepts.&quot;Seasonal influenza vaccines&quot;


/*
@input: The client has a history of anaphylactic reactions

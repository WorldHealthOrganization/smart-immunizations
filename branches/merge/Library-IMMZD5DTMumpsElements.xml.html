library IMMZD5DTMumpsElements

using FHIR version '4.0.1'
include FHIRHelpers version '4.0.1'

include WHOConcepts
include WHOCommon called WC
include WHOElements called WE

include IMMZCommon called Common
include IMMZConcepts called Concepts
include IMMZElements called Elements

context Patient

/*
@internal: Draft Medication Request for Mumps dose
*/
define &quot;Draft Medication Request for Mumps dose&quot;:
  Elements.&quot;Draft Medication Request for Patient&quot; MR 
    where MR.medication in Concepts.&quot;Mumps-containing vaccines&quot;

/*
@input: The client is currently pregnant
@pseudocode: &quot;Potential contraindications&quot; = &quot;Currently pregnant&quot;
@decision: Potential contraindications
*/
define &quot;The client is currently pregnant&quot;:
  Elements.&quot;The client is pregnant&quot;

/*
@input: The client has allergy to vaccine components
@pseudocode: '&quot;Potential contraindications&quot; = &quot;Severe allergic reactions&quot;
@decision: Potential contraindications
*/
define &quot;The client has allergy to vaccine components&quot;:
  Elements.&quot;The client has history of severe allergic reactions&quot;

/*
@input: The client has immune deficiency
@pseudocode: &quot;Potential contraindications&quot; = &quot;Immunodeficiency syndromes&quot;
@decision: Potential contraindications
*/
define &quot;The client has immune deficiency&quot;:
  Elements.&quot;The client has immunodeficiency syndromes&quot;

@input: The client is severely immunosuppressed

*/

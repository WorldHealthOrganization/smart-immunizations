/*
 * Library: IMMZD2DTRabiesPrEPLogic (IMMZ.D2.DT.Rabies)
 * Rule: Determine if the client is due for a rabies vaccination according to the national immunization schedule
 * Decision Table: Pre-exposure prophylaxis (PrEP), 2-dose scheme; high risk of rabies virus exposure
 * Trigger: IMMZ.D2 Determine required vaccination(s) if any
 */
library IMMZD2DTRabiesPrEPLogic

using FHIR version '4.0.1'
include FHIRHelpers version '4.0.1'

include IMMZD2DTRabiesEncounterElements called Encounter

parameter Today Date default Today()

context Patient


/*
@dynamicValue: Guidance
*/
define &quot;Guidance&quot;:
  case
    when &quot;Client is due for rabies vaccination&quot; then &quot;Client is due for rabies vaccination Guidance&quot;
    when &quot;Client is not due for rabies vaccination&quot; then &quot;Client is not due for rabies vaccination Guidance&quot;
    when &quot;Rabies immunization schedule is complete&quot; then &quot;Rabies immunization schedule is complete Guidance&quot;
    else ''
  end

define &quot;Has Guidance&quot;:
  &quot;Guidance&quot; is not null and &quot;Guidance&quot; != ''

/*
@output: Client is due for rabies vaccination Case 1
@pseudocode: &quot;Immunization recommendation status&quot; = &quot;Due&quot;
*/
define &quot;Client is due for rabies vaccination Case 1&quot;:
  Encounter.&quot;No rabies primary series doses were administered&quot;

/*
@output: Client is due for rabies vaccination Case 2
@pseudocode: &quot;Immunization recommendation status&quot; = &quot;Due&quot;
*/
define &quot;Client is due for rabies vaccination Case 2&quot;:
  Encounter.&quot;One rabies primary series doses were administered&quot;
    and Encounter.&quot;The latest rabies dose was administered more than 7 days ago&quot;

/*
@output: Client is due for rabies vaccination
@pseudocode: &quot;Immunization recommendation status&quot; = &quot;Due&quot;
*/
define &quot;Client is due for rabies vaccination&quot;:
  &quot;Client is due for rabies vaccination Case 1&quot;
    or &quot;Client is due for rabies vaccination Case 2&quot;

/*
@output: Client is due for rabies vaccination Guidance
@guidance: Should vaccinate client with first rabies dose as no rabies doses were administered and client is at high risk of exposure. 
Check for contraindications.
@guidance: Should vaccinate client with second rabies dose as the latest rabies dose was administerd 7 or more days ago. 
Check for contraindications.
*/
define &quot;Client is due for rabies vaccination Guidance&quot;:
  case
    when &quot;Client is due for rabies vaccination Case 1&quot; then 'Should vaccinate client with first rabies dose as no rabies doses were administered and client is at high risk of exposure. 
Check for contraindications.'
    when &quot;Client is due for rabies vaccination Case 2&quot; then 'Should vaccinate client with second rabies dose as the latest rabies dose was administerd 7 or more days ago. 
Check for contraindications.'
    else ''
  end

/*
@output: Client is not due for rabies vaccination
@pseudocode: &quot;Immunization recommendation status&quot; = &quot;Not due&quot;
*/
define &quot;Client is not due for rabies vaccination&quot;:
  Encounter.&quot;One rabies primary series doses were administered&quot;
    and Encounter.&quot;The latest rabies dose was administered less than 7 days ago&quot;

/*
@output: Client is not due for rabies vaccination Guidance
@guidance: Should not vaccinate client with second rabies dose as the latest rabies dose was administered less than 7 days ago. 
Check for any other vaccines due and inform the caregiver of when to come back for the second dose.
*/
define &quot;Client is not due for rabies vaccination Guidance&quot;:
  'Should not vaccinate client with second rabies dose as the latest rabies dose was administered less than 7 days ago. 
Check for any other vaccines due and inform the caregiver of when to come back for the second dose.'

/*
@output: Rabies immunization schedule is complete
@pseudocode: &quot;Immunization recommendation status&quot; = &quot;Complete&quot;
*/
define &quot;Rabies immunization schedule is complete&quot;:
  Encounter.&quot;Two rabies primary series doses were administered&quot;

/*
@output: Rabies immunization schedule is complete Guidance
@guidance: Rabies immunization schedule is complete. Two rabies primary series doses were administered. 
Check for any other vaccines due.
*/
define &quot;Rabies immunization schedule is complete Guidance&quot;:
  'Rabies immunization schedule is complete. Two rabies primary series doses were administered. 
Check for any other vaccines due.'


/*
@test: Test expected results based on example patients

Check for contraindications.'
    when Patient.id = '09.12' then &quot;Client is not due for rabies vaccination&quot; and &quot;Guidance&quot; = 'Should not vaccinate client with second rabies dose as the latest rabies dose was administered less than 7 days ago. 
Check for any other vaccines due and inform the caregiver of when to come back for the second dose.'
    when Patient.id = '10.12' then &quot;Client is due for rabies vaccination Case 2&quot; and &quot;Guidance&quot; = 'Should vaccinate client with second rabies dose as the latest rabies dose was administerd 7 or more days ago. 
Check for contraindications.'
    when Patient.id = '11.1' then &quot;Rabies immunization schedule is complete&quot; and &quot;Guidance&quot; = 'Rabies immunization schedule is complete. Two rabies primary series doses were administered. 
Check for any other vaccines due.'
    else 'No test case set'
  end

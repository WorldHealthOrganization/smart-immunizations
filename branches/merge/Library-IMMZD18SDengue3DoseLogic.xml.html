/*
 * Library: IMMZD18SDengue3DoseLogic (IMMZ.D18.S.Dengue schedule)
 * Schedule Table: CYD-TDV (Dengvaxia), 3-dose schedule
 */
library IMMZD18SDengue3DoseLogic

using FHIR version '4.0.1'
include FHIRHelpers version '4.0.1'

include WHOCommon called WC

include IMMZCommon called Common
include IMMZConcepts called Concepts

include IMMZEncounterElements called IE
include IMMZD2DTDengueEncounterElements called Encounter

parameter Today Date default Today()

context Patient


/*
@output: Dengue dose 1
@description: Provision of dengue dose 1
@trigger: No dengue doses from primary series were administered
Count of vaccines administered (where &quot;Vaccine type&quot; = &quot;Dengue vaccines&quot; and &quot;Type of dose&quot; = &quot;Primary series&quot;) = 0
@triggerDate: To be determined by Member States based on vaccination strategy
*/
define &quot;Dengue dose 1&quot;:
  // Member state determines if this is necessary
  Encounter.&quot;Client's age is between 9 years and 45 years&quot;
  and &quot;Dengue dose 1 Expiration&quot; &gt; Today
  and Encounter.&quot;No dengue primary series doses were administered&quot;
  and not &quot;First dengue dose from the primary series was administered&quot;

/*
@output: Dengue dose 1 Create
@create: Dengue dose 1 should be provided based on vaccination strategy
*/
define &quot;Dengue dose 1 Create&quot;:
  if &quot;Dengue dose 1&quot; 
  then 'Dengue dose 1 should be provided based on vaccination strategy' + '
Due Date: ' + ToString(&quot;Dengue dose 1 Due Date&quot;) + '
Expiration: ' + ToString(&quot;Dengue dose 1 Expiration&quot;)
  else ''

/*
@dynamicValue: Dengue dose 1 Due Date
@pseudocode: &quot;Date of birth&quot; + 9 years
*/
define &quot;Dengue dose 1 Due Date&quot;:
  if &quot;Dengue dose 1&quot; then Patient.birthDate + 9 years
  else null

/*
@dynamicValue: Dengue dose 1 Overdue
@pseudocode: To be determined by Member States
*/
define &quot;Dengue dose 1 Overdue&quot;:
  null

/*
@dynamicValue: Dengue dose 1 Expiration
@pseudocode: &quot;Date of birth&quot; + 46 years
*/
define &quot;Dengue dose 1 Expiration&quot;:
  Patient.birthDate + 46 years
  
/*
@complete: First dengue dose from the primary series was administered
@pseudocode: Count of vaccines administered (where &quot;Vaccine type&quot; = &quot;Dengue vaccines&quot; and &quot;Type of dose&quot; = &quot;Primary series&quot;) = 1
*/
define &quot;First dengue dose from the primary series was administered&quot;:
  Encounter.&quot;One dengue primary series dose was administered&quot;

/*
@output: Dengue dose 2
@description: Provision of the dengue dose 2
@trigger: First dengue dose from the primary series was administered
Count of vaccines administered (where &quot;Vaccine type&quot; = &quot;Dengue vaccines&quot; and &quot;Type of dose&quot; = &quot;Primary series&quot;) = 1
@triggerDate: &quot;Date and time of vaccination&quot; (where &quot;Vaccine type&quot; = &quot;Dengue vaccines&quot;)
*/
define &quot;Dengue dose 2&quot;:
  &quot;Dengue dose 2 Expiration&quot; &gt; Today
  and &quot;First dengue dose from the primary series was administered&quot;
  and not &quot;Second dengue dose from the primary series was administered&quot;

/*
@output: Dengue dose 2 Create
@create: Dengue dose 2 should be provided if the previous dose was provide more than 6 months ago
*/
define &quot;Dengue dose 2 Create&quot;:
  if &quot;Dengue dose 2&quot; 
  then 'Dengue dose 2 should be provided if the previous dose was provide more than 6 months ago' + '
Due Date: ' + ToString(&quot;Dengue dose 2 Due Date&quot;) + '
Expiration: ' + ToString(&quot;Dengue dose 2 Expiration&quot;)
  else ''

@dynamicValue: Dengue dose 2 Due Date

*/
define &quot;Dengue dose 2 Due Date&quot;:
  if &quot;Dengue dose 2&quot; then Encounter.&quot;Date of Latest Dengue Dose&quot; + 6 months
  else null

/*
@dynamicValue: Dengue dose 2 Overdue
@pseudocode: To be determined by Member States
*/
define &quot;Dengue dose 2 Overdue&quot;:
  null

/*
@dynamicValue: Dengue dose 2 Expiration
@pseudocode: &quot;Date of birth&quot; + 46 years
*/
define &quot;Dengue dose 2 Expiration&quot;:
  Patient.birthDate + 46 years
  
/*
@complete: Second dengue dose from the primary series was administered
@pseudocode: Count of vaccines administered (where &quot;Vaccine type&quot; = &quot;Dengue vaccines&quot; and &quot;Type of dose&quot; = &quot;Primary series&quot;) = 2
*/
define &quot;Second dengue dose from the primary series was administered&quot;:
  Encounter.&quot;Two dengue primary series doses were administered&quot;

/*
@output: Dengue dose 3
@description: Provision of dengue dose 3
@trigger: Second dengue dose from the primary series was administered
Count of vaccines administered (where &quot;Vaccine type&quot; = &quot;Dengue vaccines&quot; and &quot;Type of dose&quot; = &quot;Primary series&quot;) = 2
@triggerDate: &quot;Date and time of vaccination&quot; (where &quot;Vaccine type&quot; = &quot;Dengue vaccines&quot;)
*/
define &quot;Dengue dose 3&quot;:
  &quot;Dengue dose 2 Expiration&quot; &gt; Today
  and &quot;Second dengue dose from the primary series was administered&quot;
  and not &quot;Third dengue dose from the primary series was administered. The primary series has been completed&quot;

/*
@output: Dengue dose 3 Create
@create: Dengue dose 3 should be provided if the previous dose was provide more than 6 months ago

Due Date: ' + ToString(&quot;Dengue dose 3 Due Date&quot;) + '

  else null

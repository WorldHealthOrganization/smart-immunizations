/*
  * Library: IMMZD2DTHPVElements
  */
library IMMZD2DTHPVElements

using FHIR version '4.0.1'
include FHIRHelpers version '4.0.1'

include WHOConcepts
include WHOCommon called WC
include WHOElements called WE

include IMMZCommon called Common
include IMMZConcepts called Concepts
include IMMZElements called Elements


context Patient

/*
@internal: HPV containing Doses Administered to Patient
*/
define &quot;HPV Doses Administered to Patient&quot;:
  Elements.&quot;Doses Administered to Patient&quot; I
  where
    I.vaccineCode in Concepts.&quot;HPV vaccines&quot;

/*
@internal: HPV containing Doses Administered to Patient that are in the Primary series
*/
define &quot;HPV Primary Series Doses Administered to Patient&quot;:
  &quot;HPV Doses Administered to Patient&quot;.seriesPrimary()

/*
@internal: Number of HPV Primary Series doses
*/
define &quot;Number of HPV Primary Series Doses Administered&quot;:
  Count(&quot;HPV Primary Series Doses Administered to Patient&quot;)

/*
@input: Client's age is less than 9 years
@pseudocode: Today's date − &quot;Date of birth&quot; &lt; 9 years
@code: Client's age is less than 9 years-40
@decision: IMMZ.D2.DT.HPV.2 doses: 2-dose schedule
@decision: IMMZ.D2.DT.HPV.Single dose: Alternative single-dose schedule
*/
define &quot;Client's age is less than 9 years&quot;:
  Elements.&quot;Current Patient Age In Years&quot; &lt; 9

/*
@input: No HPV primary series doses were administered
@pseudocode: Count of vaccines administered (where &quot;Vaccine type&quot; = &quot;HPV vaccines&quot; and &quot;Type of dose&quot; = &quot;Primary series&quot;) = 0
@code: No HPV primary series doses were administered-112
@decision: IMMZ.D2.DT.HPV.2 doses: 2-dose schedule
@decision: IMMZ.D2.DT.HPV.Single dose: Alternative single-dose schedule
*/
define &quot;No HPV primary series doses were administered&quot;:
  &quot;Number of HPV Primary Series Doses Administered&quot; = 0

/*
@input: Client's age is between 9 and 14 years
@pseudocode: 9 years ≤ Today's date − &quot;Date of birth&quot; ≤ 14 years
@code: Client's age is between 9 and 14 years-51
@decision: IMMZ.D2.DT.HPV.2 doses: 2-dose schedule
*/
define &quot;Client's age is between 9 and 14 years&quot;:
  9 &lt;= Elements.&quot;Current Patient Age In Years&quot;
  and Elements.&quot;Current Patient Age In Years&quot; &lt;= 14

/*
@input: Client's biological sex is female
@pseudocode: &quot;Sex&quot; = &quot;Female&quot;
@code: Client's biological sex is female-16
@decision: IMMZ.D2.DT.HPV.2 doses: 2-dose schedule
*/
define &quot;Client's biological sex is female&quot;:
  Patient.gender = 'female'

/*
@input: Client's age more than or equal to 15 years
@pseudocode: Today's date − &quot;Date of birth&quot; ≥ 15 years
@code: Client's age more than or equal to 15 years-41
@decision: IMMZ.D2.DT.HPV.2 doses: 2-dose schedule
*/
define &quot;Client's age more than or equal to 15 years&quot;:
  Elements.&quot;Current Patient Age In Years&quot; &gt;= 15

/*
@input: Client's biological sex is not female
@pseudocode: &quot;Sex&quot; ≠ &quot;Female&quot;
@code: Client's biological sex is not female-16
@decision: IMMZ.D2.DT.HPV.2 doses: 2-dose schedule
*/
define &quot;Client's biological sex is not female&quot;:
  not &quot;Client's biological sex is female&quot;

/*
@input: One HPV primary series dose was administered
@pseudocode: Count of vaccines administered (where &quot;Vaccine type&quot; = &quot;HPV vaccines&quot; and &quot;Type of dose&quot; = &quot;Primary series&quot;) = 1
@code: One HPV primary series dose was administered-112
@decision: IMMZ.D2.DT.HPV.2 doses: 2-dose schedule
@decision: IMMZ.D2.DT.HPV.Single dose: Alternative single-dose schedule
*/
define &quot;One HPV primary series dose was administered&quot;:
  &quot;Number of HPV Primary Series Doses Administered&quot; = 1

/*
@input: The latest HPV dose was administered less than 6 months ago
@pseudocode: Today's date − latest &quot;Date and time of vaccination&quot; (where &quot;Vaccine type&quot; = &quot;HPV vaccines&quot;) &lt; 6 months
@code: The latest HPV dose was administered less than 6 months ago-103
@decision: IMMZ.D2.DT.HPV.2 doses: 2-dose schedule
@decision: IMMZ.D2.DT.HPV.Single dose: Alternative single-dose schedule
*/
define &quot;The latest HPV dose was administered less than 6 months ago&quot;:
  &quot;Date of Latest HPV Dose&quot; is not null
    and duration in months between &quot;Date of Latest HPV Dose&quot; and Now() &lt; 6

/*
@input: The latest HPV dose was administered more than 6 months ago
@pseudocode: Today's date − latest &quot;Date and time of vaccination&quot; (where &quot;Vaccine type&quot; = &quot;HPV vaccines&quot;) ≥ 6 months
@code: The latest HPV dose was administered more than 6 months ago-103
@decision: IMMZ.D2.DT.HPV.2 doses: 2-dose schedule
@decision: IMMZ.D2.DT.HPV.Single dose: Alternative single-dose schedule
*/
define &quot;The latest HPV dose was administered more than 6 months ago&quot;:
  not &quot;The latest HPV dose was administered less than 6 months ago&quot;

/*
@input: Two HPV primary series doses were administered
@pseudocode: Count of vaccines administered (where &quot;Vaccine type&quot; = &quot;HPV vaccines&quot; and &quot;Type of dose&quot; = &quot;Primary series&quot;) = 2
@code: Two HPV primary series doses were administered-112
@decision: IMMZ.D2.DT.HPV.2 doses: 2-dose schedule
@decision: IMMZ.D2.DT.HPV.Single dose: Alternative single-dose schedule
*/
define &quot;Two HPV primary series doses were administered&quot;:
  &quot;Number of HPV Primary Series Doses Administered&quot; = 2

/*
@input: Client's HIV status is positive
@pseudocode: &quot;HIV status&quot; = &quot;HIV-positive&quot;
@code: Client's HIV status is positive-29
@decision: IMMZ.D2.DT.HPV.2 doses: 2-dose schedule
@decision: IMMZ.D2.DT.HPV.Single dose: Alternative single-dose schedule
*/
define &quot;Client's HIV status is positive&quot;:
  Elements.&quot;Client's HIV status is positive&quot;

/*
@input: Client is immunocompromised
@pseudocode: &quot;Immunocompromised&quot; = TRUE
@code: Client is immunocompromised-26
@decision: IMMZ.D2.DT.HPV.2 doses: 2-dose schedule
@decision: IMMZ.D2.DT.HPV.Single dose: Alternative single-dose schedule
*/
define &quot;Client is immunocompromised&quot;:
  Elements.&quot;Client is immunocompromised&quot;

/*
@input: Client is not immunocompromised
@pseudocode: &quot;Immunocompromised&quot; ≠ TRUE
@code: Client is not immunocompromised-26
@decision: IMMZ.D2.DT.HPV.2 doses: 2-dose schedule
@decision: IMMZ.D2.DT.HPV.Single dose: Alternative single-dose schedule
*/
define &quot;Client is not immunocompromised&quot;:
  Elements.&quot;Client is not immunocompromised&quot;

/*
@input: Client's HIV status is negative or unknown
@pseudocode: &quot;HIV status&quot; ≠ &quot;HIV-positive&quot;
@code: Client's HIV status is negative or unknown-29
@decision: IMMZ.D2.DT.HPV.2 doses: 2-dose schedule
@decision: IMMZ.D2.DT.HPV.Single dose: Alternative single-dose schedule
*/
define &quot;Client's HIV status is negative or unknown&quot;:
  Elements.&quot;Client's HIV status is negative or unknown&quot;

/*
@input: Three HPV primary series doses were administered
@pseudocode: Count of vaccines administered (where &quot;Vaccine type&quot; = &quot;HPV vaccines&quot; and &quot;Type of dose&quot; = &quot;Primary series&quot;) = 3
@code: Three HPV primary series doses were administered-112
@decision: IMMZ.D2.DT.HPV.2 doses: 2-dose schedule
@decision: IMMZ.D2.DT.HPV.Single dose: Alternative single-dose schedule
*/
define &quot;Three HPV primary series doses were administered&quot;:
  &quot;Number of HPV Primary Series Doses Administered&quot; = 3

/*
@input: Client's age is between 9 and 20 years
@pseudocode: 9 years ≤ Today's date − &quot;Date of birth&quot; ≤ 20 years
@code: Client's age is between 9 and 20 years-51
@decision: IMMZ.D2.DT.HPV.Single dose: Alternative single-dose schedule
*/
define &quot;Client's age is between 9 and 20 years&quot;:
  9 &lt;= Elements.&quot;Current Patient Age In Years&quot;
  and Elements.&quot;Current Patient Age In Years&quot; &lt;= 20

/*
@input: Client's age is more than 20 years
@pseudocode: Today's date − &quot;Date of birth&quot; &gt; 20 years
@code: Client's age is more than 20 years-41
@decision: IMMZ.D2.DT.HPV.Single dose: Alternative single-dose schedule
*/
define &quot;Client's age is more than 20 years&quot;:
  Elements.&quot;Current Patient Age In Years&quot; &gt; 20

/*
@internal: Date of Latest HPV Dose
*/
define &quot;Date of Latest HPV Dose&quot;:
  date from start of &quot;HPV Doses Administered to Patient&quot;.mostRecent().occurrence.toInterval()

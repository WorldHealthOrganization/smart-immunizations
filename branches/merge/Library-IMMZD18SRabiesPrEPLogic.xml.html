/*
 * Library: IMMZD18SRabiesPrEPLogic (IMMZ.D18.S.Rabies schedule)
 * Schedule Table: Rabies schedule (2-dose scheme)
 */
library IMMZD18SRabiesPrEPLogic

using FHIR version '4.0.1'
include FHIRHelpers version '4.0.1'

include WHOCommon called WC

include IMMZCommon called Common
include IMMZConcepts called Concepts

include IMMZEncounterElements called IE
include IMMZD2DTRabiesEncounterElements called Encounter

parameter Today Date default Today()

context Patient


/*
@output: Rabies dose 1
@description: Provision of rabies dose 1
@trigger: To be determined by Member States
@triggerDate: To be determined by Member States
*/
define &quot;Rabies dose 1&quot;:
  // Member state must define the trigger for if the first dose is needed.
  Encounter.&quot;No rabies primary series doses were administered&quot;
  and not &quot;First rabies dose from the primary series was administered&quot;

/*
@output: Rabies dose 1 Create
@create: Rabies dose 1 should be provided if the client is at a high risk of rabies virus exposure
*/
define &quot;Rabies dose 1 Create&quot;:
  if &quot;Rabies dose 1&quot; 
  then 'Rabies dose 1 should be provided if the client is at a high risk of rabies virus exposure'
  else ''

/*
@dynamicValue: Rabies dose 1 Due Date
@pseudocode: To be determined by Member States
*/
define &quot;Rabies dose 1 Due Date&quot;:
  null

/*
@dynamicValue: Rabies dose 1 Overdue
@pseudocode: To be determined by Member States
*/
define &quot;Rabies dose 1 Overdue&quot;:
  null

/*
@dynamicValue: Rabies dose 1 Expiration
@pseudocode: To be determined by Member States
*/
define &quot;Rabies dose 1 Expiration&quot;:
  null
  
/*
@complete: First rabies dose from the primary series was administered
@pseudocode: Count of vaccines administered (where &quot;Vaccine type&quot; = &quot;Rabies vaccines&quot; and &quot;Type of dose&quot; = &quot;Primary series&quot;) = 1
*/
define &quot;First rabies dose from the primary series was administered&quot;:
  Encounter.&quot;One rabies primary series doses were administered&quot;

/*
@output: Rabies dose 2
@description: Provision of rabies dose 2
@trigger: First rabies dose from the primary series was administered
Count of vaccines administered (where &quot;Vaccine type&quot; = &quot;Rabies vaccines&quot; and &quot;Type of dose&quot; = &quot;Primary series&quot;) = 1
@triggerDate: &quot;Date and time of vaccination&quot; (where &quot;Vaccine type&quot; = &quot;Rabies vaccines&quot;)
*/
define &quot;Rabies dose 2&quot;:
  &quot;First rabies dose from the primary series was administered&quot;
  and not &quot;Second rabies dose from the primary series was administered. The primary series has been completed&quot;

/*
@output: Rabies dose 2 Create
@create: Rabies dose 2 should be provided if previous dose was given more than 1 week ago
*/
define &quot;Rabies dose 2 Create&quot;:
  if &quot;Rabies dose 2&quot; 
  then 'Rabies dose 2 should be provided if previous dose was given more than 1 week ago' + '
Due Date: ' + ToString(&quot;Rabies dose 2 Due Date&quot;)
  else ''

/*
@dynamicValue: Rabies dose 2 Due Date
@pseudocode: &quot;Date and time of vaccination&quot; (where &quot;Vaccine type&quot; = &quot;Rabies vaccines&quot;) + 7 days
*/
define &quot;Rabies dose 2 Due Date&quot;:
  if &quot;Rabies dose 2&quot; then Encounter.&quot;Date of Latest Rabies Dose&quot; + 7 days
  else null

/*
@dynamicValue: Rabies dose 2 Overdue
@pseudocode: To be determined by Member States
*/
define &quot;Rabies dose 2 Overdue&quot;:
  null

/*
@dynamicValue: Rabies dose 2 Expiration
@pseudocode: To be determined by Member States
*/
define &quot;Rabies dose 2 Expiration&quot;:
  null
  
/*
@complete: Second rabies dose from the primary series was administered. The primary series has been completed
@pseudocode: &quot;Completed the primary vaccination series&quot; = TRUE (where &quot;Vaccine type&quot; = &quot;Rabies vaccines&quot;)
*/
define &quot;Second rabies dose from the primary series was administered. The primary series has been completed&quot;:
  Encounter.&quot;Two rabies primary series doses were administered&quot;

    when Patient.id = '10.12' then &quot;Rabies dose 2&quot;
    when Patient.id = '11.1' then &quot;Second rabies dose from the primary series was administered. The primary series has been completed&quot;
    else 'No test case set'
  end

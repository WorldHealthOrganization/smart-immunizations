/*
 * Library: IMMZD18SMeaslesSupplementaryLogic (IMMZ.D18.S.Measles.Supplementary dose schedule)
 * Schedule Table: Measles supplementary dose schedule
 */
library IMMZD18SMeaslesSupplementaryLogic

using FHIR version '4.0.1'
include FHIRHelpers version '4.0.1'

include IMMZD2DTMeaslesEncounterElements called Encounter

parameter Today Date default Today()

context Patient


/*
@output: Measles-containing vaccine (MCV) supplementary dose
@description: Provision of a MCV supplementary dose
@trigger: Routine measles immunization schedule is complete
&quot;Completed the primary vaccination series&quot; = TRUE (where &quot;Vaccine type&quot; = &quot;Measles-containing vaccines&quot;) AND Member States defined logic for supplementary dose applied
@triggerDate: &quot;Date when the primary vaccination series was  completed&quot;
*/
define &quot;Measles-containing vaccine (MCV) supplementary dose&quot;:
  not &quot;MCV supplementary dose was administered&quot; and Encounter.&quot;Measles routine immunization schedule is complete&quot;

/*
@output: Measles-containing vaccine (MCV) supplementary dose Create
@create: Child is due for a MCV supplementary dose if child is HIV-positive, on antiretroviral therapy (ART) and immune reconstitution has been achieved.
*/
define &quot;Measles-containing vaccine (MCV) supplementary dose Create&quot;:
  if &quot;Measles-containing vaccine (MCV) supplementary dose&quot; 
  then 'Child is due for a MCV supplementary dose if child is HIV-positive, on antiretroviral therapy (ART) and immune reconstitution has been achieved.' + '
Due Date: ' + ToString(&quot;Measles-containing vaccine (MCV) supplementary dose Due Date&quot;)
  else ''

/*
@dynamicValue: Measles-containing vaccine (MCV) supplementary dose Due Date
@pseudocode: &quot;Date when the primary vaccination series completed&quot; + 4 weeks
*/
define &quot;Measles-containing vaccine (MCV) supplementary dose Due Date&quot;:
  if &quot;Measles-containing vaccine (MCV) supplementary dose&quot; then Encounter.&quot;Date when the primary vaccination series completed&quot; + 4 weeks
  else null

/*
@dynamicValue: Measles-containing vaccine (MCV) supplementary dose Overdue
@pseudocode: To be determined by Member States; however, there is no recommended overdue date and individuals are always eligible to be vaccinated.
*/
define &quot;Measles-containing vaccine (MCV) supplementary dose Overdue&quot;:
  null

/*
@dynamicValue: Measles-containing vaccine (MCV) supplementary dose Expiration
@pseudocode: To be determined by Member States; however, there is no recommended expiration date and individuals are always eligible to be vaccinated.
*/
define &quot;Measles-containing vaccine (MCV) supplementary dose Expiration&quot;:
  null

/*
@complete: MCV supplementary dose was administered
@pseudocode: Count of vaccines administered (where &quot;Vaccine type&quot; = &quot;Measles-containing vaccines&quot; and &quot;Type of dose&quot; = &quot;Supplementary dose&quot;) = 1
*/
define &quot;MCV supplementary dose was administered&quot;:
  Encounter.&quot;Measles supplementary dose was administered&quot;


/*
@test: Test expected results based on example patients
*/
define &quot;Test Validation&quot;:
  case
    when Patient.id = '47.PS2' then &quot;Measles-containing vaccine (MCV) supplementary dose&quot;
    when Patient.id = '47.D1' then not &quot;Measles-containing vaccine (MCV) supplementary dose&quot;
    else 'No test case set'
  end

 * Library: IMMZD2DTCholeraWCRBSVaccine2DosesLogic (IMMZ.D2.DT.Cholera.WC-rBS vaccine 2 doses)
 * Rule: Determine if the client is due for a cholera vaccination according to the national immunization schedule
 * Decision Table: Whole cell-recombinant B subunit (WC-rBS) vaccine, 2-dose schedule for clients aged over 5 years
 * Trigger: IMMZ.D2 Determine required vaccination(s) if any
 */
library IMMZD2DTCholeraWCRBSVaccine2DosesLogic

using FHIR version '4.0.1'
include FHIRHelpers version '4.0.1'

include IMMZD2DTCholeraEncounterElements called Encounter

parameter Today Date default Today()

context Patient


/*
@dynamicValue: Guidance
*/
define &quot;Guidance&quot;:
  case 
    when &quot;Client is due for cholera vaccination&quot; then &quot;Client is due for cholera vaccination Guidance&quot;
    when &quot;Client is not due for cholera vaccination&quot; then &quot;Client is not due for cholera vaccination Guidance&quot;
    when &quot;Primary series is complete. Client is due for cholera booster dose&quot; then &quot;Primary series is complete. Client is due for cholera booster dose Guidance&quot; 
    else ''
  end
  
define &quot;Has Guidance&quot;:
  &quot;Guidance&quot; is not null and &quot;Guidance&quot; != ''

@output: Client is due for cholera vaccination Case 1
@pseudocode: &quot;Immunization recommendation status&quot; = &quot;Due&quot;
*/

  Encounter.&quot;Client's age is more than 5 years&quot;
    and Encounter.&quot;No cholera primary series dose was administered&quot;

/*
@output: Client is due for cholera vaccination Case 2
@pseudocode: &quot;Immunization recommendation status&quot; = &quot;Due&quot;
*/

  Encounter.&quot;Client's age is more than 5 years&quot;
    and Encounter.&quot;At least one cholera primary series dose was administered&quot;
    and Encounter.&quot;Dose number of the latest primary series dose is one&quot;
    and Encounter.&quot;The latest cholera dose was administered more than 1 week and less than 6 weeks ago&quot;

/*
@output: Client is due for cholera vaccination Case 3
@pseudocode: &quot;Immunization recommendation status&quot; = &quot;Due&quot;
&quot;Completed the booster series&quot; (where &quot;Vaccine type&quot; = &quot;Cholera vaccines&quot;) = FALSE when dose 1 is administered
*/
define &quot;Client is due for cholera vaccination Case 3&quot;:
  Encounter.&quot;Client's age is more than 5 years&quot;
    and Encounter.&quot;At least one cholera primary series dose was administered&quot;
    and Encounter.&quot;Dose number of the latest primary series dose is one&quot;
    and Encounter.&quot;The latest cholera dose was administered more than 6 weeks ago&quot;

/*
@output: Client is due for cholera vaccination Case 4
@pseudocode: &quot;Immunization recommendation status&quot; = &quot;Due&quot;
&quot;Completed the primary vaccination series&quot; (where &quot;Vaccine type&quot; = &quot;Cholera vaccines&quot;) = FALSE when dose 1 is administered
*/
define &quot;Client is due for cholera vaccination Case 4&quot;:
  Encounter.&quot;Client's age is more than 5 years&quot;
    and Encounter.&quot;At least one cholera primary series dose was administered&quot;
    and Encounter.&quot;Dose number of the latest primary series dose is more than or equal to two&quot;
    and Encounter.&quot;The latest cholera dose was administered approximately more than 2 years ago {Member States defined interval}&quot;

@output: Client is due for cholera vaccination

*/

  &quot;Client is due for cholera vaccination Case 1&quot;
    or &quot;Client is due for cholera vaccination Case 2&quot;
    or &quot;Client is due for cholera vaccination Case 3&quot;
    or &quot;Client is due for cholera vaccination Case 4&quot;

/*
@output: Client is due for cholera vaccination Guidance
@guidance = Should vaccinate client with cholera dose as no cholera doses have been administered.
Check for contraindications.
@guidance = Should vaccinate client with cholera dose as the latest cholera dose was administered in the last 1 to 6 weeks.

@guidance = Should vaccinate client with cholera dose and restart primary series as the latest cholera dose was administered more than 6 weeks ago.

Check for contraindications.
*/

  case
    when &quot;Client is due for cholera vaccination Case 1&quot; then 'Should vaccinate client with cholera dose as no cholera doses have been administered.
Check for contraindications.'
    when &quot;Client is due for cholera vaccination Case 2&quot; then 'Should vaccinate client with cholera dose as the latest cholera dose was administered in the last 1 to 6 weeks.
Check for contraindications.'
    when &quot;Client is due for cholera vaccination Case 3&quot; then 'Should vaccinate client with cholera dose and restart primary series as the latest cholera dose was administered more than 6 weeks ago.
Check for contraindications.'
    when &quot;Client is due for cholera vaccination Case 4&quot; then 'Should vaccinate client with cholera dose and restart primary series as the latest cholera dose was administered approximately more than 2 years ago. 
Check for contraindications.'
    else ''
  end

    and Encounter.&quot;Dose number of the latest primary series dose is one&quot;

define &quot;Primary series is complete. Client is due for cholera booster dose&quot;:

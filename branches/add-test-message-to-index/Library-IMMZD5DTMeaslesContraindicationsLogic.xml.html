 * Library: IMMZD5DTMeaslesContraindicationsLogic (IMMZ.D5.DT.Measles contraindications)
 * Rule: Check for contraindications before administering the vaccine(s) due
 * Decision Table: IMMZ.D5.DT.Measles contraindications
 * Trigger: IMMZ.D5 Determine vaccine(s) to be administered based on contraindications
 */
library IMMZD5DTMeaslesContraindicationsLogic

using FHIR version '4.0.1'
include FHIRHelpers version '4.0.1'

include IMMZD5DTMeaslesEncounterElements called Encounter

parameter Today Date default Today()

context Patient

/*
  @dynamicValue: Draft Medication Request ID for Measles dose
  */
  define &quot;Draft Medication Request ID for Measles dose&quot;:
    First(Encounter.&quot;Draft Medication Request for Measles dose&quot;).id


/*
@dynamicValue: Guidance
*/
define &quot;Guidance&quot;:
  case 
    when &quot;Measles vaccination contraindicated&quot; then &quot;Measles vaccination contraindicated Guidance&quot;
    when &quot;Measles vaccination could be contraindicated. Clinical judgement required. Create a clinical note&quot; then &quot;Measles vaccination could be contraindicated. Clinical judgement required. Create a clinical note Guidance&quot;
    when &quot;Clinical judgement required. Create a clinical note&quot; then &quot;Clinical judgement required. Create a clinical note Guidance&quot; 
    else ''
  end
  
define &quot;Has Guidance&quot;:
  &quot;Guidance&quot; is not null and &quot;Guidance&quot; != ''

/*
@output: Measles vaccination contraindicated Case 1
@pseudocode: &quot;Immunization recommendation status&quot; = &quot;Contraindicated&quot;
*/
define &quot;Measles vaccination contraindicated Case 1&quot;:
  Encounter.&quot;The client is pregnant&quot;

/*
@output: Measles vaccination contraindicated Case 2
@pseudocode: &quot;Immunization recommendation status&quot; = &quot;Contraindicated&quot;
*/
define &quot;Measles vaccination contraindicated Case 2&quot;:
  Encounter.&quot;The client is severely immunosuppressed&quot;

/*
@output: Measles vaccination contraindicated
@pseudocode: &quot;Immunization recommendation status&quot; = &quot;Contraindicated&quot;
*/
define &quot;Measles vaccination contraindicated&quot;:
  &quot;Measles vaccination contraindicated Case 1&quot;
    or &quot;Measles vaccination contraindicated Case 2&quot;

/*
@output: Measles vaccination contraindicated Guidance
@guidance = Do not vaccinate client with measles as measles vaccination is contraindicated for pregnant client.
@guidance = Do not vaccinate client with measles as measles vaccination is contraindicated for severely immunosuppressed client.
*/
define &quot;Measles vaccination contraindicated Guidance&quot;:
  case
    when &quot;Measles vaccination contraindicated Case 1&quot; then 'Do not vaccinate client with measles as measles vaccination is contraindicated for pregnant client.'
    when &quot;Measles vaccination contraindicated Case 2&quot; then 'Do not vaccinate client with measles as measles vaccination is contraindicated for severely immunosuppressed client.'
    else ''
  end

/*
@output: Measles vaccination could be contraindicated. Clinical judgement required. Create a clinical note
@pseudocode: &quot;Immunization recommendation status&quot; = &quot;Further evaluation needed&quot;
*/
define &quot;Measles vaccination could be contraindicated. Clinical judgement required. Create a clinical note&quot;:
  Encounter.&quot;The client has history of anaphylactic reactions&quot;

/*
@output: Measles vaccination could be contraindicated. Clinical judgement required. Create a clinical note Guidance
@guidance: Do not vaccinate client with measles if client has history of anaphylactic reaction to any component of the vaccine (e.g. neomycin or gelatin).
*/
define &quot;Measles vaccination could be contraindicated. Clinical judgement required. Create a clinical note Guidance&quot;:
  'Do not vaccinate client with measles if client has history of anaphylactic reaction to any component of the vaccine (e.g. neomycin or gelatin).'

/*
@output: Clinical judgement required. Create a clinical note Case 1
@pseudocode: &quot;Immunization recommendation status&quot; = &quot;Further evaluation needed&quot;
*/
define &quot;Clinical judgement required. Create a clinical note Case 1&quot;:
  Encounter.&quot;The client has history of severe allergic reactions&quot;

/*
@output: Clinical judgement required. Create a clinical note Case 2
@pseudocode: &quot;Immunization recommendation status&quot; = &quot;Further evaluation needed&quot;
*/
define &quot;Clinical judgement required. Create a clinical note Case 2&quot;:
  Encounter.&quot;The client has a symptomatic HIV infection&quot;

/*
@output: Clinical judgement required. Create a clinical note
@pseudocode: &quot;Immunization recommendation status&quot; = &quot;Further evaluation needed&quot;
*/
define &quot;Clinical judgement required. Create a clinical note&quot;:
  &quot;Clinical judgement required. Create a clinical note Case 1&quot;
    or &quot;Clinical judgement required. Create a clinical note Case 2&quot;

/*
@output: Clinical judgement required. Create a clinical note Guidance
@guidance = Do not vaccinate client with measles if client has history of severe allergic reactions to any component of the vaccine (e.g. neomycin or gelatin).
@guidance = Client has symptomatic HIV infection. Measles vaccination may be considered if the client is not severely immunosuppressed according to conventional definitions, consider risks of vaccination and make a clinical judgement.
*/
define &quot;Clinical judgement required. Create a clinical note Guidance&quot;:
  case
    when &quot;Clinical judgement required. Create a clinical note Case 1&quot; then 'Do not vaccinate client with measles if client has history of severe allergic reactions to any component of the vaccine (e.g. neomycin or gelatin).'
    when &quot;Clinical judgement required. Create a clinical note Case 2&quot; then 'Client has symptomatic HIV infection. Measles vaccination may be considered if the client is not severely immunosuppressed according to conventional definitions, consider risks of vaccination and make a clinical judgement.'
    else ''
  end

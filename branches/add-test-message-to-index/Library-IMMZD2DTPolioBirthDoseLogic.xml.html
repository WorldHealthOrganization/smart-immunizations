 * Library: IMMZD2DTPolioBirthDoseLogic (IMMZ.D2.DT.Polio.Birth dose)
 * Rule: Determine if the client is due for a poliovirus vaccination according to the national immunization schedule
 * Decision Table: Birth dose administration as part of &quot;bivalent oral polio vaccine (bOPV) plus inactivated polio vaccine (IPV)&quot; schedule (in countries that are poliomyelitis (polio)-endemic countries or at high risk of importation and subsequent spread of polio that follow the combined bOPV–IPV schedule)
 * Trigger: IMMZ.D2 Determine required vaccination(s) if any
 */
library IMMZD2DTPolioBirthDoseLogic

using FHIR version '4.0.1'
include FHIRHelpers version '4.0.1'

include IMMZD2DTPolioEncounterElements called Encounter

parameter Today Date default Today()

context Patient


/*
@dynamicValue: Guidance
*/
define &quot;Guidance&quot;:
  case 
    when &quot;Client is due for the bOPV birth dose&quot; then &quot;Client is due for the bOPV birth dose Guidance&quot;
    when &quot;Client is not due for the bOPV birth dose&quot; then &quot;Client is not due for the bOPV birth dose Guidance&quot; 
    else ''
  end
  
define &quot;Has Guidance&quot;:
  &quot;Guidance&quot; is not null and &quot;Guidance&quot; != ''

/*
@output: Client is due for the bOPV birth dose
@pseudocode: &quot;Immunization recommendation status&quot; = &quot;Due&quot; (where &quot;Type of poliovirus dose&quot; = &quot;bOPV&quot;)
*/
define &quot;Client is due for the bOPV birth dose&quot;:
  Encounter.&quot;bOPV birth dose (a zero dose) was not administered&quot;
    and Encounter.&quot;Client's age is less than 1 week&quot;

/*
@output: Client is due for the bOPV birth dose Guidance
@guidance: Should vaccinate client with bOPV birth dose (a zero dose) as poliovirus birth dose was not administered and client's age is less than 1 week. Check for contraindications.
*/
define &quot;Client is due for the bOPV birth dose Guidance&quot;:
  'Should vaccinate client with bOPV birth dose (a zero dose) as poliovirus birth dose was not administered and client\'s age is less than 1 week. Check for contraindications.'

/*
@output: Client is not due for the bOPV birth dose Case 1
@pseudocode: &quot;Immunization recommendation status&quot; = &quot;Not due&quot;
*/
define &quot;Client is not due for the bOPV birth dose Case 1&quot;:
  Encounter.&quot;bOPV birth dose (a zero dose) was not administered&quot;
    and Encounter.&quot;Client's age is more than or equal to 1 week&quot;

/*
@output: Client is not due for the bOPV birth dose Case 2
@pseudocode: &quot;Immunization recommendation status&quot; = &quot;Not due&quot;
*/
define &quot;Client is not due for the bOPV birth dose Case 2&quot;:
  Encounter.&quot;bOPV birth dose (a zero dose) was administered&quot;

/*
@output: Client is not due for the bOPV birth dose
@pseudocode: &quot;Immunization recommendation status&quot; = &quot;Not due&quot;
*/
define &quot;Client is not due for the bOPV birth dose&quot;:
  &quot;Client is not due for the bOPV birth dose Case 1&quot;
    or &quot;Client is not due for the bOPV birth dose Case 2&quot;

/*
@output: Client is not due for the bOPV birth dose Guidance
@guidance = Should not vaccinate client with bOPV birth dose (a zero dose) as client's age is more than or equal to 1 week. Check for any other vaccines due and inform the caregiver of when to come back for the next dose.
@guidance = Poliovirus birth dose was already administered. Check poliovirus immunization schedule.
*/
define &quot;Client is not due for the bOPV birth dose Guidance&quot;:
  case
    when &quot;Client is not due for the bOPV birth dose Case 1&quot; then 'Should not vaccinate client with bOPV birth dose (a zero dose) as client\'s age is more than or equal to 1 week. Check for any other vaccines due and inform the caregiver of when to come back for the next dose.'
    when &quot;Client is not due for the bOPV birth dose Case 2&quot; then 'Poliovirus birth dose was already administered. Check poliovirus immunization schedule.'
    else ''
  end


/*
@test: Test expected results based on example patients
*/

    when Patient.id = 'Polio09.2' then &quot;Client is not due for the bOPV birth dose Case 1&quot; and &quot;Guidance&quot; = 'Should not vaccinate client with bOPV birth dose (a zero dose) as client\'s age is more than or equal to 1 week. Check for any other vaccines due and inform the caregiver of when to come back for the next dose.'
    when Patient.id = 'Polio10.1' then &quot;Client is not due for the bOPV birth dose Case 2&quot; and &quot;Guidance&quot; = 'Poliovirus birth dose was already administered. Check poliovirus immunization schedule.'
    else 'No test case set'
  end

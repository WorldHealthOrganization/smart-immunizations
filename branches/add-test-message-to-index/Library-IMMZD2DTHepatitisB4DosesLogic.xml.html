 * Trigger: IMMZ.D2 Determine required vaccination(s) if any
 */

    when &quot;Hepatitis B immunization schedule is complete&quot; then &quot;Hepatitis B immunization schedule is complete Guidance&quot; 
    else ''
  end

    or &quot;Client is not due for a hepatitis B vaccination Case 3&quot;

    when &quot;Client is not due for a hepatitis B vaccination Case 3&quot; then 'Should not vaccinate client with third hepatitis B dose as the latest Hepatitis B dose was administered less than 4 weeks ago.
Check for any other vaccines due, and inform the caregiver of when to come back for the third dose.'
    else ''
  end
